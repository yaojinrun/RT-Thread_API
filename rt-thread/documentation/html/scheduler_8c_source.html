88  RT_ASSERT(thread != RT_NULL);
89 
90  if (*((rt_uint8_t *)thread->stack_addr) != '#' ||
91  (rt_uint32_t)thread->sp <= (rt_uint32_t)thread->stack_addr ||
92  (rt_uint32_t)thread->sp >
93  (rt_uint32_t)thread->stack_addr + (rt_uint32_t)thread->stack_size)
94  {
95  rt_uint32_t level;
96 
97  rt_kprintf("thread:%s stack overflow\n", thread->name);

100  extern long list_thread(void);
101  list_thread();
102  }
103 #endif
104  level = rt_hw_interrupt_disable();
105  while (level);
106  }

158 {

161 
162 #if RT_THREAD_PRIORITY_MAX > 32

176  rt_current_thread = to_thread;

190 /**
191  * @brief 线程调度函数
192  *
193  * 该函数将执行一次调度。它将选择一个具有最高优先级的线程，然后切换运行。
194  */
195 void rt_schedule(void)
196 {
197  rt_base_t level;
198  struct rt_thread *to_thread;
199  struct rt_thread *from_thread;
200 
201  /* disable interrupt */
202  level = rt_hw_interrupt_disable();
203 
204  /* check the scheduler is enabled or not */
205  if (rt_scheduler_lock_nest == 0)
206  {
207  register rt_ubase_t highest_ready_priority;
208 
209 #if RT_THREAD_PRIORITY_MAX <= 32
210  highest_ready_priority = __rt_ffs(rt_thread_ready_priority_group) - 1;
211 #else
212  register rt_ubase_t number;
213 
214  number = __rt_ffs(rt_thread_ready_priority_group) - 1;
215  highest_ready_priority = (number << 3) + __rt_ffs(rt_thread_ready_table[number]) - 1;
216 #endif
217 
218  /* get switch to thread */
219  to_thread = rt_list_entry(rt_thread_priority_table[highest_ready_priority].next,
220  struct rt_thread,
221  tlist);
222 
223  /* if the destination thread is not the same as current thread */
224  if (to_thread != rt_current_thread)
225  {
226  rt_current_priority = (rt_uint8_t)highest_ready_priority;
227  from_thread = rt_current_thread;
228  rt_current_thread = to_thread;
229 
230  RT_OBJECT_HOOK_CALL(rt_scheduler_hook, (from_thread, to_thread));
231 
232  /* switch to new thread */
233  RT_DEBUG_LOG(RT_DEBUG_SCHEDULER,
234  ("[%d]switch to priority#%d "
235  "thread:%.*s(sp:0x%p), "
236  "from thread:%.*s(sp: 0x%p)\n",
237  rt_interrupt_nest, highest_ready_priority,
238  RT_NAME_MAX, to_thread->name, to_thread->sp,
239  RT_NAME_MAX, from_thread->name, from_thread->sp));
240 
241 #ifdef RT_USING_OVERFLOW_CHECK
242  _rt_scheduler_stack_check(to_thread);
243 #endif
244 
245  if (rt_interrupt_nest == 0)
246  {
247  extern void rt_thread_handle_sig(rt_bool_t clean_state);
248 
249  rt_hw_context_switch((rt_uint32_t)&from_thread->sp,
250  (rt_uint32_t)&to_thread->sp);
251 
252  /* enable interrupt */
253  rt_hw_interrupt_enable(level);
254 
255 #ifdef RT_USING_SIGNALS
256  /* check signal status */
257  rt_thread_handle_sig(RT_TRUE);
258 #endif
259  }
260  else
261  {
262  RT_DEBUG_LOG(RT_DEBUG_SCHEDULER, ("switch in interrupt\n"));
263 
264  rt_hw_context_switch_interrupt((rt_uint32_t)&from_thread->sp,
265  (rt_uint32_t)&to_thread->sp);
266  /* enable interrupt */
267  rt_hw_interrupt_enable(level);
268  }
269  }
270  else
271  {
272  /* enable interrupt */
273  rt_hw_interrupt_enable(level);
274  }
275  }
276  else
277  {
278  /* enable interrupt */
279  rt_hw_interrupt_enable(level);
280  }
281 }

283 /*
284  * @brief 就绪线程插入函数
285  *
286  * 该函数将把一个线程插入系统就绪队列。线程的状态将被设置为READY并从挂起队列中删除。
287  *
288  * @param thread 将被插入的线程
289  * @note 请不要再用户应用中调用此函数。
290  */
291 void rt_schedule_insert_thread(struct rt_thread *thread)
292 {
293  register rt_base_t temp;
294 
295  RT_ASSERT(thread != RT_NULL);
296 
297  /* disable interrupt */
298  temp = rt_hw_interrupt_disable();
299 
300  /* change stat */
301  thread->stat = RT_THREAD_READY | (thread->stat & ~RT_THREAD_STAT_MASK);
302 
303  /* insert thread to ready list */
304  rt_list_insert_before(&(rt_thread_priority_table[thread->current_priority]),
305  &(thread->tlist));
306 
307  /* set priority mask */
308 #if RT_THREAD_PRIORITY_MAX <= 32
309  RT_DEBUG_LOG(RT_DEBUG_SCHEDULER, ("insert thread[%.*s], the priority: %d\n",
310  RT_NAME_MAX, thread->name, thread->current_priority));
311 #else
312  RT_DEBUG_LOG(RT_DEBUG_SCHEDULER,
313  ("insert thread[%.*s], the priority: %d 0x%x %d\n",
314  RT_NAME_MAX,
315  thread->name,
316  thread->number,
317  thread->number_mask,
318  thread->high_mask));
319 #endif
320 
321 #if RT_THREAD_PRIORITY_MAX > 32
322  rt_thread_ready_table[thread->number] |= thread->high_mask;
323 #endif
324  rt_thread_ready_priority_group |= thread->number_mask;
325 
326  /* enable interrupt */
327  rt_hw_interrupt_enable(temp);
328 }

331  * @brief
332  *
333  * 该函数将把一个线程从系统就绪队列中删除。

335  * @param thread 将被移除的线程
336  *
337  * @note 请不要再用户应用中调用此函数。

339 void rt_schedule_remove_thread(struct rt_thread *thread)
340 {
341  register rt_base_t temp;
342 
343  RT_ASSERT(thread != RT_NULL);
344 
345  /* disable interrupt */
346  temp = rt_hw_interrupt_disable();
347 
348 #if RT_THREAD_PRIORITY_MAX <= 32
349  RT_DEBUG_LOG(RT_DEBUG_SCHEDULER, ("remove thread[%.*s], the priority: %d\n",
350  RT_NAME_MAX, thread->name,
351  thread->current_priority));
352 #else
353  RT_DEBUG_LOG(RT_DEBUG_SCHEDULER,
354  ("remove thread[%.*s], the priority: %d 0x%x %d\n",
355  RT_NAME_MAX,
356  thread->name,
357  thread->number,
358  thread->number_mask,
359  thread->high_mask));
360 #endif
361 
362  /* remove thread from ready list */
363  rt_list_remove(&(thread->tlist));
364  if (rt_list_isempty(&(rt_thread_priority_table[thread->current_priority])))
365  {
366 #if RT_THREAD_PRIORITY_MAX > 32
367  rt_thread_ready_table[thread->number] &= ~thread->high_mask;
368  if (rt_thread_ready_table[thread->number] == 0)
369  {
370  rt_thread_ready_priority_group &= ~thread->number_mask;
371  }
372 #else
373  rt_thread_ready_priority_group &= ~thread->number_mask;
374 #endif
375  }
376 
377  /* enable interrupt */
378  rt_hw_interrupt_enable(temp);
379 }
