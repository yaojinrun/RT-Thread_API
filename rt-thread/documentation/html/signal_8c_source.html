92 /*
93  * To deliver a signal to thread, there are cases:
94  * 1. When thread is suspended, function resumes thread and
95  * set signal stat;
96  * 2. When thread is ready:
97  * - If function delivers a signal to self thread, just handle
98  * it.
99  * - If function delivers a signal to another ready thread, OS
100  * should build a slice context to handle it.
101  */
102 static void _signal_deliver(rt_thread_t tid)
103 {
104  rt_ubase_t level;
105 
106  /* thread is not interested in pended signals */
107  if (!(tid->sig_pending & tid->sig_mask)) return;
108 
109  level = rt_hw_interrupt_disable();
110  if ((tid->stat & RT_THREAD_STAT_MASK) == RT_THREAD_SUSPEND)
111  {
112  /* resume thread to handle signal */
113  rt_thread_resume(tid);
114  /* add signal state */
115  tid->stat |= RT_THREAD_STAT_SIGNAL;
116 
117  rt_hw_interrupt_enable(level);
118 
119  /* re-schedule */
120  rt_schedule();
121  }
122  else
123  {
124  if (tid == rt_thread_self())
125  {
126  /* add signal state */
127  tid->stat |= RT_THREAD_STAT_SIGNAL;
128 
129  rt_hw_interrupt_enable(level);
130 
131  /* do signal action in self thread context */
132  rt_thread_handle_sig(RT_TRUE);
133  }
134  else if (!((tid->stat & RT_THREAD_STAT_MASK) & RT_THREAD_STAT_SIGNAL))
135  {
136  /* add signal state */
137  tid->stat |= RT_THREAD_STAT_SIGNAL;
138 
139  /* point to the signal handle entry */
140  tid->sig_ret = tid->sp;
141  tid->sp = rt_hw_stack_init((void *)_signal_entry, RT_NULL,
142  (void *)((char *)tid->sig_ret - 32), RT_NULL);
143 
144  rt_hw_interrupt_enable(level);
145  dbg_log(DBG_LOG, "signal stack pointer @ 0x%08x\n", tid->sp);
146 
147  /* re-schedule */
148  rt_schedule();
149  }
150  else
151  {
152  rt_hw_interrupt_enable(level);
153  }
154  }
155 }

157 /**
158  * @ingroup Signal
159  * @brief 信号安装
160  *
161  * 如果线程要处理某一信号，那么就要在线程中安装该信号。安装信号主要
162  * 用来确定信号值及线程针对该信号值的动作之间的映射关系，即线程将要处理
163  * 哪个信号；该信号被传递给线程时，将执行何种操作。
164  *
165  * @param signo 信号值（只有 SIGUSR1 和 SIGUSR2 是开放给用户使用的）
166  * @param handler 信号值的处理。可以忽略该信号（参数设为 SIG_IGN）；
167  * 可以采用系统默认方式处理信号 (参数设为 SIG_DFL)；也可以自己实现
168  * 处理方式 (参数指定一个函数地址)。
169  *
170  * @note 成功安装信号前的 handler 值，失败返回SIG_ERR。
171  */
172 rt_sighandler_t rt_signal_install(int signo, rt_sighandler_t handler)
173 {
174  rt_sighandler_t old;
175  rt_thread_t tid = rt_thread_self();
176 
177  if (!sig_valid(signo)) return SIG_ERR;
178 
179  rt_enter_critical();
180  if (tid->sig_vectors == RT_NULL)
181  {
182  rt_thread_alloc_sig(tid);
183  }
184 
185  if (tid->sig_vectors)
186  {
187  old = tid->sig_vectors[signo];
188 
189  if (handler == SIG_IGN) tid->sig_vectors[signo] = RT_NULL;
190  else if (handler == SIG_DFL) tid->sig_vectors[signo] = _signal_default_handler;
191  else tid->sig_vectors[signo] = handler;
192  }
193  rt_exit_critical();
194 
195  return old;
196 }

249 /**
250  * @ingroup Signal
251  * @brief 等待信号
252  *
253  * 该函数将等待 set 信号的到来，如果没有等到这个信号，则把线程给挂起来，
254  * 直到等到这个信号或者等待时间超过指定的超时时间 timeout。
255  *
256  * @param set 指定等待的信号值
257  * @param si 等待的信号句柄
258  * @param timeout 指定的超时时间
259  *
260  */
261 int rt_signal_wait(const rt_sigset_t *set, rt_siginfo_t *si, rt_int32_t timeout)
262 {
263  int ret = RT_EOK;
264  rt_base_t level;
265  rt_thread_t tid = rt_thread_self();
266  struct siginfo_node *si_node = RT_NULL, *si_prev = RT_NULL;
267 
268  /* current context checking */
269  RT_DEBUG_IN_THREAD_CONTEXT;
270 
271  /* parameters check */
272  if (set == NULL || *set == 0 || si == NULL )
273  {
274  ret = -RT_EINVAL;
275  goto __done_return;
276  }
277 
278  /* clear siginfo to avoid unknown value */
279  memset(si, 0x0, sizeof(rt_siginfo_t));
280 
281  level = rt_hw_interrupt_disable();
282 
283  /* already pending */
284  if (tid->sig_pending & *set) goto __done;
285 
286  if (timeout == 0)
287  {
288  ret = -RT_ETIMEOUT;
289  goto __done_int;
290  }
291 
292  /* suspend self thread */
293  rt_thread_suspend(tid);
294  /* set thread stat as waiting for signal */
295  tid->stat |= RT_THREAD_STAT_SIGNAL_WAIT;
296 
297  /* start timeout timer */
298  if (timeout != RT_WAITING_FOREVER)
299  {
300  /* reset the timeout of thread timer and start it */
301  rt_timer_control(&(tid->thread_timer),
302  RT_TIMER_CTRL_SET_TIME,
303  &timeout);
304  rt_timer_start(&(tid->thread_timer));
305  }
306  rt_hw_interrupt_enable(level);
307 
308  /* do thread scheduling */
309  rt_schedule();
310 
311  level = rt_hw_interrupt_disable();
312 
313  /* remove signal waiting flag */
314  tid->stat &= ~RT_THREAD_STAT_SIGNAL_WAIT;
315 
316  /* check errno of thread */
317  if (tid->error == -RT_ETIMEOUT)
318  {
319  tid->error = RT_EOK;
320  rt_hw_interrupt_enable(level);
321 
322  /* timer timeout */
323  ret = -RT_ETIMEOUT;
324  goto __done_return;
325  }
326 
327 __done:
328  /* to get the first matched pending signals */
329  si_node = (struct siginfo_node *)tid->si_list;
330  while (si_node)
331  {
332  int signo;
333 
334  signo = si_node->si.si_signo;
335  if (sig_mask(signo) & *set)
336  {
337  *si = si_node->si;
338 
339  dbg_log(DBG_LOG, "sigwait: %d sig raised!\n", signo);
340  if (si_prev) si_prev->list.next = si_node->list.next;
341  else tid->si_list = si_node->list.next;
342 
343  /* clear pending */
344  tid->sig_pending &= ~sig_mask(signo);
345  rt_mp_free(si_node);
346  break;
347  }
348 
349  si_prev = si_node;
350  si_node = (void *)rt_slist_entry(si_node->list.next, struct siginfo_node, list);
351  }
352 
353 __done_int:
354  rt_hw_interrupt_enable(level);
355 
356 __done_return:
357  return ret;
358 }

360 void rt_thread_handle_sig(rt_bool_t clean_state)
361 {
362  rt_base_t level;
363 
364  rt_thread_t tid = rt_thread_self();
365  struct siginfo_node *si_node;
366 
367  level = rt_hw_interrupt_disable();
368  if (tid->sig_pending & tid->sig_mask)
369  {
370  /* if thread is not waiting for signal */
371  if (!(tid->stat & RT_THREAD_STAT_SIGNAL_WAIT))
372  {
373  while (tid->sig_pending & tid->sig_mask)
374  {
375  int signo, error;
376  rt_sighandler_t handler;
377 
378  si_node = (struct siginfo_node *)tid->si_list;
379  if (!si_node) break;
380 
381  /* remove this sig info node from list */
382  if (si_node->list.next == RT_NULL)
383  tid->si_list = RT_NULL;
384  else
385  tid->si_list = (void *)rt_slist_entry(si_node->list.next, struct siginfo_node, list);
386 
387  signo = si_node->si.si_signo;
388  handler = tid->sig_vectors[signo];
389  rt_hw_interrupt_enable(level);
390 
391  dbg_log(DBG_LOG, "handle signal: %d, handler 0x%08x\n", signo, handler);
392  if (handler) handler(signo);
393 
394  level = rt_hw_interrupt_disable();
395  tid->sig_pending &= ~sig_mask(signo);
396  error = -RT_EINTR;
397 
398  rt_mp_free(si_node); /* release this siginfo node */
399  /* set errno in thread tcb */
400  tid->error = error;
401  }
402 
403  /* whether clean signal status */
404  if (clean_state == RT_TRUE) tid->stat &= ~RT_THREAD_STAT_SIGNAL;
405  }
406  }
407 
408  rt_hw_interrupt_enable(level);
409 }

444  if (si_list)

458  }

466 /**
467  * @ingroup Signal
468  * @brief 信号的发送
469  *
470  * 调用该函数可以用来向任何线程发送任何信号。
471  *
472  * @param tid 接收信号的线程
473  * @param sig 指定的信号值
474  *
475  */
476 int rt_thread_kill(rt_thread_t tid, int sig)
477 {
478  siginfo_t si;
479  rt_base_t level;
480  struct siginfo_node *si_node;
481 
482  RT_ASSERT(tid != RT_NULL);
483  if (!sig_valid(sig)) return -RT_EINVAL;
484 
485  dbg_log(DBG_INFO, "send signal: %d\n", sig);
486  si.si_signo = sig;
487  si.si_code = SI_USER;
488  si.si_value.sival_ptr = RT_NULL;
489 
490  level = rt_hw_interrupt_disable();
491  if (tid->sig_pending & sig_mask(sig))
492  {
493  /* whether already emits this signal? */
494  struct rt_slist_node *node;
495  struct siginfo_node *entry;
496 
497  node = (struct rt_slist_node *)tid->si_list;
498  rt_hw_interrupt_enable(level);
499 
500  /* update sig info */
501  rt_enter_critical();
502  for (; (node) != RT_NULL; node = node->next)
503  {
504  entry = rt_slist_entry(node, struct siginfo_node, list);
505  if (entry->si.si_signo == sig)
506  {
507  memcpy(&(entry->si), &si, sizeof(siginfo_t));
508  rt_exit_critical();
509  return 0;
510  }
511  }
512  rt_exit_critical();
513 
514  /* disable interrupt to protect tcb */
515  level = rt_hw_interrupt_disable();
516  }
517  else
518  {
519  /* a new signal */
520  tid->sig_pending |= sig_mask(sig);
521  }
522  rt_hw_interrupt_enable(level);
523 
524  si_node = (struct siginfo_node *) rt_mp_alloc(_rt_siginfo_pool, 0);
525  if (si_node)
526  {
527  rt_slist_init(&(si_node->list));
528  memcpy(&(si_node->si), &si, sizeof(siginfo_t));
529 
530  level = rt_hw_interrupt_disable();
531  if (!tid->si_list) tid->si_list = si_node;
532  else
533  {
534  struct siginfo_node *si_list;
535 
536  si_list = (struct siginfo_node *)tid->si_list;
537  rt_slist_append(&(si_list->list), &(si_node->list));
538  }
539  rt_hw_interrupt_enable(level);
540  }
541  else
542  {
543  dbg_log(DBG_ERROR, "The allocation of signal info node failed.\n");
544  }
545 
546  /* deliver signal to this thread */
547  _signal_deliver(tid);
548 
549  return RT_EOK;
550 }
