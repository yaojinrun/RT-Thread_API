296 /**
297  * @brief 启动定时器
298  *
299  * 当定时器被创建或者初始化以后，并不会被立即启动，必须在调用该函数接口后，才开始工作。
300  *
301  * @param timer 将要被启动的定时器
302  *
303  * @return 成功 RT_EOK ; 错误则返回 -RT_ERROR
304  */
305 rt_err_t rt_timer_start(rt_timer_t timer)
306 {
307  unsigned int row_lvl;
308  rt_list_t *timer_list;
309  register rt_base_t level;
310  rt_list_t *row_head[RT_TIMER_SKIP_LIST_LEVEL];
311  unsigned int tst_nr;
312  static unsigned int random_nr;
313 
314  /* timer check */
315  RT_ASSERT(timer != RT_NULL);
316 
317  /* stop timer firstly */
318  level = rt_hw_interrupt_disable();
319  /* remove timer from list */
320  _rt_timer_remove(timer);
321  /* change status of timer */
322  timer->parent.flag &= ~RT_TIMER_FLAG_ACTIVATED;
323  rt_hw_interrupt_enable(level);
324 
325  RT_OBJECT_HOOK_CALL(rt_object_take_hook, (&(timer->parent)));
326 
327  /*
328  * get timeout tick,
329  * the max timeout tick shall not great than RT_TICK_MAX/2
330  */
331  RT_ASSERT(timer->init_tick < RT_TICK_MAX / 2);
332  timer->timeout_tick = rt_tick_get() + timer->init_tick;
333 
334  /* disable interrupt */
335  level = rt_hw_interrupt_disable();
336 
337 #ifdef RT_USING_TIMER_SOFT
338  if (timer->parent.flag & RT_TIMER_FLAG_SOFT_TIMER)
339  {
340  /* insert timer to soft timer list */
341  timer_list = rt_soft_timer_list;
342  }
343  else
344 #endif
345  {
346  /* insert timer to system timer list */
347  timer_list = rt_timer_list;
348  }
349 
350  row_head[0] = &timer_list[0];
351  for (row_lvl = 0; row_lvl < RT_TIMER_SKIP_LIST_LEVEL; row_lvl++)
352  {
353  for (; row_head[row_lvl] != timer_list[row_lvl].prev;
354  row_head[row_lvl] = row_head[row_lvl]->next)
355  {
356  struct rt_timer *t;
357  rt_list_t *p = row_head[row_lvl]->next;
358 
359  /* fix up the entry pointer */
360  t = rt_list_entry(p, struct rt_timer, row[row_lvl]);
361 
362  /* If we have two timers that timeout at the same time, it's
363  * preferred that the timer inserted early get called early.
364  * So insert the new timer to the end the the some-timeout timer
365  * list.
366  */
367  if ((t->timeout_tick - timer->timeout_tick) == 0)
368  {
369  continue;
370  }
371  else if ((t->timeout_tick - timer->timeout_tick) < RT_TICK_MAX / 2)
372  {
373  break;
374  }
375  }
376  if (row_lvl != RT_TIMER_SKIP_LIST_LEVEL - 1)
377  row_head[row_lvl + 1] = row_head[row_lvl] + 1;
378  }
379 
380  /* Interestingly, this super simple timer insert counter works very very
381  * well on distributing the list height uniformly. By means of "very very
382  * well", I mean it beats the randomness of timer->timeout_tick very easily
383  * (actually, the timeout_tick is not random and easy to be attacked). */
384  random_nr++;
385  tst_nr = random_nr;
386 
387  rt_list_insert_after(row_head[RT_TIMER_SKIP_LIST_LEVEL - 1],
388  &(timer->row[RT_TIMER_SKIP_LIST_LEVEL - 1]));
389  for (row_lvl = 2; row_lvl <= RT_TIMER_SKIP_LIST_LEVEL; row_lvl++)
390  {
391  if (!(tst_nr & RT_TIMER_SKIP_LIST_MASK))
392  rt_list_insert_after(row_head[RT_TIMER_SKIP_LIST_LEVEL - row_lvl],
393  &(timer->row[RT_TIMER_SKIP_LIST_LEVEL - row_lvl]));
394  else
395  break;
396  /* Shift over the bits we have tested. Works well with 1 bit and 2
397  * bits. */
398  tst_nr >>= (RT_TIMER_SKIP_LIST_MASK + 1) >> 1;
399  }
400 
401  timer->parent.flag |= RT_TIMER_FLAG_ACTIVATED;
402 
403  /* enable interrupt */
404  rt_hw_interrupt_enable(level);
405 
406 #ifdef RT_USING_TIMER_SOFT
407  if (timer->parent.flag & RT_TIMER_FLAG_SOFT_TIMER)
408  {
409  /* check whether timer thread is ready */
410  if ((timer_thread.stat & RT_THREAD_STAT_MASK) != RT_THREAD_READY)
411  {
412  /* resume timer thread to check soft timer */
413  rt_thread_resume(&timer_thread);
414  rt_schedule();
415  }
416  }
417 #endif
418 
419  return -RT_EOK;
420 }
421 RTM_EXPORT(rt_timer_start);

497 /*
498  * 该函数将检查定时器的链表，如果有定时器超时事件发生，那么相应的超时函数将被运行。
499  *
500  * @note 该函数应该在系统定时器中断中调用
501  */
502 void rt_timer_check(void)
503 {
504  struct rt_timer *t;
505  rt_tick_t current_tick;
506  register rt_base_t level;
507 
508  RT_DEBUG_LOG(RT_DEBUG_TIMER, ("timer check enter\n"));
509 
510  current_tick = rt_tick_get();
511 
512  /* disable interrupt */
513  level = rt_hw_interrupt_disable();
514 
515  while (!rt_list_isempty(&rt_timer_list[RT_TIMER_SKIP_LIST_LEVEL - 1]))
516  {
517  t = rt_list_entry(rt_timer_list[RT_TIMER_SKIP_LIST_LEVEL - 1].next,
518  struct rt_timer, row[RT_TIMER_SKIP_LIST_LEVEL - 1]);
519 
520  /*
521  * It supposes that the new tick shall less than the half duration of
522  * tick max.
523  */
524  if ((current_tick - t->timeout_tick) < RT_TICK_MAX / 2)
525  {
526  RT_OBJECT_HOOK_CALL(rt_timer_timeout_hook, (t));
527 
528  /* remove timer from timer list firstly */
529  _rt_timer_remove(t);
530 
531  /* call timeout function */
532  t->timeout_func(t->parameter);
533 
534  /* re-get tick */
535  current_tick = rt_tick_get();
536 
537  RT_DEBUG_LOG(RT_DEBUG_TIMER, ("current tick: %d\n", current_tick));
538 
539  if ((t->parent.flag & RT_TIMER_FLAG_PERIODIC) &&
540  (t->parent.flag & RT_TIMER_FLAG_ACTIVATED))
541  {
542  /* start it */
543  t->parent.flag &= ~RT_TIMER_FLAG_ACTIVATED;
544  rt_timer_start(t);
545  }
546  else
547  {
548  /* stop timer */
549  t->parent.flag &= ~RT_TIMER_FLAG_ACTIVATED;
550  }
551  }
552  else
553  break;
554  }
555 
556  /* enable interrupt */
557  rt_hw_interrupt_enable(level);
558 
559  RT_DEBUG_LOG(RT_DEBUG_TIMER, ("timer check leave\n"));
560 }

573 /**
574  * 该函数将检查定时器的链表，如果有定时器超时事件发生，那么相应的超时函数将被运行。
575  */
576 void rt_soft_timer_check(void)
577 {
578  rt_tick_t current_tick;
579  rt_list_t *n;
580  struct rt_timer *t;
581 
582  RT_DEBUG_LOG(RT_DEBUG_TIMER, ("software timer check enter\n"));
583 
584  current_tick = rt_tick_get();
585 
586  /* lock scheduler */
587  rt_enter_critical();
588 
589  for (n = rt_soft_timer_list[RT_TIMER_SKIP_LIST_LEVEL - 1].next;
590  n != &(rt_soft_timer_list[RT_TIMER_SKIP_LIST_LEVEL - 1]);)
591  {
592  t = rt_list_entry(n, struct rt_timer, row[RT_TIMER_SKIP_LIST_LEVEL - 1]);
593 
594  /*
595  * It supposes that the new tick shall less than the half duration of
596  * tick max.
597  */
598  if ((current_tick - t->timeout_tick) < RT_TICK_MAX / 2)
599  {
600  RT_OBJECT_HOOK_CALL(rt_timer_timeout_hook, (t));
601 
602  /* move node to the next */
603  n = n->next;
604 
605  /* remove timer from timer list firstly */
606  _rt_timer_remove(t);
607 
608  /* not lock scheduler when performing timeout function */
609  rt_exit_critical();
610  /* call timeout function */
611  t->timeout_func(t->parameter);
612 
613  /* re-get tick */
614  current_tick = rt_tick_get();
615 
616  RT_DEBUG_LOG(RT_DEBUG_TIMER, ("current tick: %d\n", current_tick));
617 
618  /* lock scheduler */
619  rt_enter_critical();
620 
621  if ((t->parent.flag & RT_TIMER_FLAG_PERIODIC) &&
622  (t->parent.flag & RT_TIMER_FLAG_ACTIVATED))
623  {
624  /* start it */
625  t->parent.flag &= ~RT_TIMER_FLAG_ACTIVATED;
626  rt_timer_start(t);
627  }
628  else
629  {
630  /* stop timer */
631  t->parent.flag &= ~RT_TIMER_FLAG_ACTIVATED;
632  }
633  }
634  else break; /* not check anymore */
635  }
636 
637  /* unlock scheduler */
638  rt_exit_critical();
639 
640  RT_DEBUG_LOG(RT_DEBUG_TIMER, ("software timer check leave\n"));
641 }

643 /* system timer thread entry */
644 static void rt_thread_timer_entry(void *parameter)
645 {
646  rt_tick_t next_timeout;
647 
648  while (1)
649  {
650  /* get the next timeout tick */
651  next_timeout = rt_timer_list_next_timeout(rt_soft_timer_list);
652  if (next_timeout == RT_TICK_MAX)
653  {
654  /* no software timer exist, suspend self. */
655  rt_thread_suspend(rt_thread_self());
656  rt_schedule();
657  }
658  else
659  {
660  rt_tick_t current_tick;
661 
662  /* get current tick */
663  current_tick = rt_tick_get();
664 
665  if ((next_timeout - current_tick) < RT_TICK_MAX / 2)
666  {
667  /* get the delta timeout tick */
668  next_timeout = next_timeout - current_tick;
669  rt_thread_delay(next_timeout);
670  }
671  }
672 
673  /* check software timer */
674  rt_soft_timer_check();
675  }
676 }
