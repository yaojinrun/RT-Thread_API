234 /**
235  * @brief 初始化对象
236  *
237  * 该函数将初始化一个对象并将其放到系统对象管理器中。
238  *
239  * @param object 将要被初始化的指定对象
240  * @param type 对象类型
241  * @param name 对象的名称。在系统中对象的名称必须唯一。
242  */
243 void rt_object_init(struct rt_object *object,
244  enum rt_object_class_type type,
245  const char *name)
246 {
247  register rt_base_t temp;
248  struct rt_object_information *information;
249 
250  /* get object information */
251  information = rt_object_get_information(type);
252  RT_ASSERT(information != RT_NULL);
253 
254  /* initialize object's parameters */
255 
256  /* set object type to static */
257  object->type = type | RT_Object_Class_Static;
258 
259  /* copy name */
260  rt_strncpy(object->name, name, RT_NAME_MAX);
261 
262  RT_OBJECT_HOOK_CALL(rt_object_attach_hook, (object));
263 
264  /* lock interrupt */
265  temp = rt_hw_interrupt_disable();
266 
267  /* insert object into information object list */
268  rt_list_insert_after(&(information->object_list), &(object->list));
269 
270  /* unlock interrupt */
271  rt_hw_interrupt_enable(temp);
272 }

416 /*
417  * 该函数将从对象容器中查找具有指定名称的对象。
418  *
419  * @param name 具有指定名称额对象
420  * @param type the type of object
421  *
422  * @return 返回查找到的对象，如果没有此对象则返回 RT_NULL
423  *
424  * @note 该函数应该在中断上下文中调用。
425  */
426 rt_object_t rt_object_find(const char *name, rt_uint8_t type)
427 {
428  struct rt_object *object = RT_NULL;
429  struct rt_list_node *node = RT_NULL;
430  struct rt_object_information *information = RT_NULL;
431 
432  /* parameter check */
433  if ((name == RT_NULL) || (type > RT_Object_Class_Unknown))
434  return RT_NULL;
435 
436  /* which is invoke in interrupt status */
437  RT_DEBUG_NOT_IN_INTERRUPT;
438 
439  /* enter critical */
440  rt_enter_critical();
441 
442  /* try to find object */
443  if (information == RT_NULL)
444  {
445  information = rt_object_get_information((enum rt_object_class_type)type);
446  RT_ASSERT(information != RT_NULL);
447  }
448  for (node = information->object_list.next;
449  node != &(information->object_list);
450  node = node->next)
451  {
452  object = rt_list_entry(node, struct rt_object, list);
453  if (rt_strncmp(object->name, name, RT_NAME_MAX) == 0)
454  {
455  /* leave critical */
456  rt_exit_critical();
457 
458  return object;
459  }
460  }
461 
462  /* leave critical */
463  rt_exit_critical();
464 
465  return RT_NULL;
466 }
