71 /**
72  * @brief 中断进入
73  *
74  * 当进入中断服务程序时，BSP将调用此函数
75  *
76  * @note 请不要在程序中调用此例程。
77  *
78  * @see rt_interrupt_leave
79  */
80 void rt_interrupt_enter(void)
81 {
82  rt_base_t level;
83 
84  RT_DEBUG_LOG(RT_DEBUG_IRQ, ("irq coming..., irq nest:%d\n",
85  rt_interrupt_nest));
86 
87  level = rt_hw_interrupt_disable();
88  rt_interrupt_nest ++;
89  RT_OBJECT_HOOK_CALL(rt_interrupt_enter_hook,());
90  rt_hw_interrupt_enable(level);
91 }
92 RTM_EXPORT(rt_interrupt_enter);
93 
94 /**
95  * @brief 中断离开
96  *
97  * 当离开中断服务程序时，BSP将调用此函数
98  *
99  * @note 请不要在程序中调用此例程。
100  *
101  * @see rt_interrupt_enter
102  */
103 void rt_interrupt_leave(void)
104 {
105  rt_base_t level;
106 
107  RT_DEBUG_LOG(RT_DEBUG_IRQ, ("irq leave, irq nest:%d\n",
108  rt_interrupt_nest));
109 
110  level = rt_hw_interrupt_disable();
111  rt_interrupt_nest --;
112  RT_OBJECT_HOOK_CALL(rt_interrupt_leave_hook,());
113  rt_hw_interrupt_enable(level);
114 }
115 RTM_EXPORT(rt_interrupt_leave);
