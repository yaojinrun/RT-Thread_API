136 /**
137  * 此函数将内存的内容设置为指定的值
138  *
139  * @param s 源内存的地址
140  * @param c 将要设定的值
141  * @param count 拷贝的数据长度
142  *
143  * @return 源内存的地址
144  */
145 void *rt_memset(void *s, int c, rt_ubase_t count)
146 {
147 #ifdef RT_USING_TINY_SIZE
148  char *xs = (char *)s;
149 
150  while (count--)
151  *xs++ = c;
152 
153  return s;
154 #else
155 #define LBLOCKSIZE (sizeof(rt_int32_t))
156 #define UNALIGNED(X) ((rt_int32_t)X & (LBLOCKSIZE - 1))
157 #define TOO_SMALL(LEN) ((LEN) < LBLOCKSIZE)
158 
159  int i;
160  char *m = (char *)s;
161  rt_uint32_t buffer;
162  rt_uint32_t *aligned_addr;
163  rt_uint32_t d = c & 0xff;
164 
165  if (!TOO_SMALL(count) && !UNALIGNED(s))
166  {
167  /* If we get this far, we know that n is large and m is word-aligned. */
168  aligned_addr = (rt_uint32_t *)s;
169 
170  /* Store D into each char sized location in BUFFER so that
171  * we can set large blocks quickly.
172  */
173  if (LBLOCKSIZE == 4)
174  {
175  buffer = (d << 8) | d;
176  buffer |= (buffer << 16);
177  }
178  else
179  {
180  buffer = 0;
181  for (i = 0; i < LBLOCKSIZE; i ++)
182  buffer = (buffer << 8) | d;
183  }
184 
185  while (count >= LBLOCKSIZE * 4)
186  {
187  *aligned_addr++ = buffer;
188  *aligned_addr++ = buffer;
189  *aligned_addr++ = buffer;
190  *aligned_addr++ = buffer;
191  count -= 4 * LBLOCKSIZE;
192  }
193 
194  while (count >= LBLOCKSIZE)
195  {
196  *aligned_addr++ = buffer;
197  count -= LBLOCKSIZE;
198  }
199 
200  /* Pick up the remainder with a bytewise loop. */
201  m = (char *)aligned_addr;
202  }
203 
204  while (count--)
205  {
206  *m++ = (char)d;
207  }
208 
209  return s;
210 
211 #undef LBLOCKSIZE
212 #undef UNALIGNED
213 #undef TOO_SMALL
214 #endif
215 }
216 RTM_EXPORT(rt_memset);

218 /**
219  * 此功能将内存内容从源地址复制到目标地址。
220  *
221  * @param dst 目标内存的地址
222  * @param src 源内存的地址
223  * @param count 复制的数据长度
224  *
225  * @return 目标内存的地址
226  */
227 void *rt_memcpy(void *dst, const void *src, rt_ubase_t count)
228 {
229 #ifdef RT_USING_TINY_SIZE
230  char *tmp = (char *)dst, *s = (char *)src;
231  rt_ubase_t len;
232 
233  if (tmp <= s || tmp > (s + count))
234  {
235  while (count--)
236  *tmp ++ = *s ++;
237  }
238  else
239  {
240  for (len = count; len > 0; len --)
241  tmp[len - 1] = s[len - 1];
242  }
243 
244  return dst;
245 #else
246 
247 #define UNALIGNED(X, Y) \
248  (((rt_int32_t)X & (sizeof(rt_int32_t) - 1)) | \
249  ((rt_int32_t)Y & (sizeof(rt_int32_t) - 1)))
250 #define BIGBLOCKSIZE (sizeof(rt_int32_t) << 2)
251 #define LITTLEBLOCKSIZE (sizeof(rt_int32_t))
252 #define TOO_SMALL(LEN) ((LEN) < BIGBLOCKSIZE)
253 
254  char *dst_ptr = (char *)dst;
255  char *src_ptr = (char *)src;
256  rt_int32_t *aligned_dst;
257  rt_int32_t *aligned_src;
258  int len = count;
259 
260  /* If the size is small, or either SRC or DST is unaligned,
261  then punt into the byte copy loop. This should be rare. */
262  if (!TOO_SMALL(len) && !UNALIGNED(src_ptr, dst_ptr))
263  {
264  aligned_dst = (rt_int32_t *)dst_ptr;
265  aligned_src = (rt_int32_t *)src_ptr;
266 
267  /* Copy 4X long words at a time if possible. */
268  while (len >= BIGBLOCKSIZE)
269  {
270  *aligned_dst++ = *aligned_src++;
271  *aligned_dst++ = *aligned_src++;
272  *aligned_dst++ = *aligned_src++;
273  *aligned_dst++ = *aligned_src++;
274  len -= BIGBLOCKSIZE;
275  }
276 
277  /* Copy one long word at a time if possible. */
278  while (len >= LITTLEBLOCKSIZE)
279  {
280  *aligned_dst++ = *aligned_src++;
281  len -= LITTLEBLOCKSIZE;
282  }
283 
284  /* Pick up any residual with a byte copier. */
285  dst_ptr = (char *)aligned_dst;
286  src_ptr = (char *)aligned_src;
287  }
288 
289  while (len--)
290  *dst_ptr++ = *src_ptr++;
291 
292  return dst;
293 #undef UNALIGNED
294 #undef BIGBLOCKSIZE
295 #undef LITTLEBLOCKSIZE
296 #undef TOO_SMALL
297 #endif
298 }

311 void *rt_memmove(void *dest, const void *src, rt_ubase_t n)
312 {
313  char *tmp = (char *)dest, *s = (char *)src;
314 
315  if (s < tmp && tmp < s + n)
316  {
317  tmp += n;
318  s += n;
319 
320  while (n--)
321  *(--tmp) = *(--s);
322  }
323  else
324  {
325  while (n--)
326  *tmp++ = *s++;
327  }
328 
329  return dest;

602 #ifdef RT_PRINTF_PRECISION
603 static char *print_number(char *buf,
604  char *end,
605  long num,
606  int base,
607  int s,
608  int precision,
609  int type)
610 #else
611 static char *print_number(char *buf,
612  char *end,
613  long num,
614  int base,
615  int s,
616  int type)
617 #endif
618 {
619  char c, sign;
620 #ifdef RT_PRINTF_LONGLONG
621  char tmp[32];
622 #else
623  char tmp[16];
624 #endif
625  const char *digits;
626  static const char small_digits[] = "0123456789abcdef";
627  static const char large_digits[] = "0123456789ABCDEF";
628  register int i;
629  register int size;
630 
631  size = s;
632 
633  digits = (type & LARGE) ? large_digits : small_digits;
634  if (type & LEFT)
635  type &= ~ZEROPAD;
636 
637  c = (type & ZEROPAD) ? '0' : ' ';
638 
639  /* get sign */
640  sign = 0;
641  if (type & SIGN)
642  {
643  if (num < 0)
644  {
645  sign = '-';
646  num = -num;
647  }
648  else if (type & PLUS)
649  sign = '+';
650  else if (type & SPACE)
651  sign = ' ';
652  }
653 
654 #ifdef RT_PRINTF_SPECIAL
655  if (type & SPECIAL)
656  {
657  if (base == 16)
658  size -= 2;
659  else if (base == 8)
660  size--;
661  }
662 #endif
663 
664  i = 0;
665  if (num == 0)
666  tmp[i++] = '0';
667  else
668  {
669  while (num != 0)
670  tmp[i++] = digits[divide(&num, base)];
671  }
672 
673 #ifdef RT_PRINTF_PRECISION
674  if (i > precision)
675  precision = i;
676  size -= precision;
677 #else
678  size -= i;
679 #endif
680 
681  if (!(type & (ZEROPAD | LEFT)))
682  {
683  if ((sign) && (size > 0))
684  size--;
685 
686  while (size-- > 0)
687  {
688  if (buf <= end)
689  *buf = ' ';
690  ++ buf;
691  }
692  }
693 
694  if (sign)
695  {
696  if (buf <= end)
697  {
698  *buf = sign;
699  -- size;
700  }
701  ++ buf;
702  }
703 
704 #ifdef RT_PRINTF_SPECIAL
705  if (type & SPECIAL)
706  {
707  if (base == 8)
708  {
709  if (buf <= end)
710  *buf = '0';
711  ++ buf;
712  }
713  else if (base == 16)
714  {
715  if (buf <= end)
716  *buf = '0';
717  ++ buf;
718  if (buf <= end)
719  {
720  *buf = type & LARGE ? 'X' : 'x';
721  }
722  ++ buf;
723  }
724  }
725 #endif
726 
727  /* no align to the left */
728  if (!(type & LEFT))
729  {
730  while (size-- > 0)
731  {
732  if (buf <= end)
733  *buf = c;
734  ++ buf;
735  }
736  }
737 
738 #ifdef RT_PRINTF_PRECISION
739  while (i < precision--)
740  {
741  if (buf <= end)
742  *buf = '0';
743  ++ buf;
744  }
745 #endif
746 
747  /* put number in the temporary buffer */
748  while (i-- > 0)
749  {
750  if (buf <= end)
751  *buf = tmp[i];
752  ++ buf;
753  }
754 
755  while (size-- > 0)
756  {
757  if (buf <= end)
758  *buf = ' ';
759  ++ buf;
760  }
761 
762  return buf;
763 }
