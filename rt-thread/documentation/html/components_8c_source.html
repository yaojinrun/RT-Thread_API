228 int rtthread_startup(void)
229 {
230  rt_hw_interrupt_disable();
231 
232  /* board level initalization
233  * NOTE: please initialize heap inside board initialization.
234  */
235  rt_hw_board_init();
236 
237  /* show RT-Thread version */
238  rt_show_version();
239 
240  /* timer system initialization */
241  rt_system_timer_init();
242 
243  /* scheduler system initialization */
244  rt_system_scheduler_init();
245 
246 #ifdef RT_USING_SIGNALS
247  /* signal system initialization */
248  rt_system_signal_init();
249 #endif
250 
251  /* create init_thread */
252  rt_application_init();
253 
254  /* timer thread initialization */
255  rt_system_timer_thread_init();
256 
257  /* idle thread initialization */
258  rt_thread_idle_init();
259 
260  /* start scheduler */
261  rt_system_scheduler_start();
262 
263  /* never reach here */
264  return 0;
