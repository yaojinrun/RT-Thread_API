159 static void plug_holes(struct heap_mem *mem)
160 {
161  struct heap_mem *nmem;
162  struct heap_mem *pmem;
163 
164  RT_ASSERT((rt_uint8_t *)mem >= heap_ptr);
165  RT_ASSERT((rt_uint8_t *)mem < (rt_uint8_t *)heap_end);
166  RT_ASSERT(mem->used == 0);
167 
168  /* plug hole forward */
169  nmem = (struct heap_mem *)&heap_ptr[mem->next];
170  if (mem != nmem &&
171  nmem->used == 0 &&
172  (rt_uint8_t *)nmem != (rt_uint8_t *)heap_end)
173  {
174  /* if mem->next is unused and not end of heap_ptr,
175  * combine mem and mem->next
176  */
177  if (lfree == nmem)
178  {
179  lfree = mem;
180  }
181  mem->next = nmem->next;
182  ((struct heap_mem *)&heap_ptr[nmem->next])->prev = (rt_uint8_t *)mem - heap_ptr;
183  }
184 
185  /* plug hole backward */
186  pmem = (struct heap_mem *)&heap_ptr[mem->prev];
187  if (pmem != mem && pmem->used == 0)
188  {
189  /* if mem->prev is unused, combine mem and mem->prev */
190  if (lfree == mem)
191  {
192  lfree = pmem;
193  }
194  pmem->next = mem->next;
195  ((struct heap_mem *)&heap_ptr[mem->next])->prev = (rt_uint8_t *)pmem - heap_ptr;
196  }
197 }

199 /**
200  * @ingroup SystemInit
201  *
202  * @brief 系统内存堆的配置和初始化
203  *
204  * 该函数将初始化系统内存堆。
205  *
206  * @param begin_addr 系统内存堆的起始地址
207  * @param end_addr 系统内存堆的结束地址。
208  */
209 void rt_system_heap_init(void *begin_addr, void *end_addr)
210 {
211  struct heap_mem *mem;
212  rt_uint32_t begin_align = RT_ALIGN((rt_uint32_t)begin_addr, RT_ALIGN_SIZE);
213  rt_uint32_t end_align = RT_ALIGN_DOWN((rt_uint32_t)end_addr, RT_ALIGN_SIZE);
214 
215  RT_DEBUG_NOT_IN_INTERRUPT;
216 
217  /* alignment addr */
218  if ((end_align > (2 * SIZEOF_STRUCT_MEM)) &&
219  ((end_align - 2 * SIZEOF_STRUCT_MEM) >= begin_align))
220  {
221  /* calculate the aligned memory size */
222  mem_size_aligned = end_align - begin_align - 2 * SIZEOF_STRUCT_MEM;
223  }
224  else
225  {
226  rt_kprintf("mem init, error begin address 0x%x, and end address 0x%x\n",
227  (rt_uint32_t)begin_addr, (rt_uint32_t)end_addr);
228 
229  return;
230  }
231 
232  /* point to begin address of heap */
233  heap_ptr = (rt_uint8_t *)begin_align;
234 
235  RT_DEBUG_LOG(RT_DEBUG_MEM, ("mem init, heap begin address 0x%x, size %d\n",
236  (rt_uint32_t)heap_ptr, mem_size_aligned));
237 
238  /* initialize the start of the heap */
239  mem = (struct heap_mem *)heap_ptr;
240  mem->magic = HEAP_MAGIC;
241  mem->next = mem_size_aligned + SIZEOF_STRUCT_MEM;
242  mem->prev = 0;
243  mem->used = 0;
244 #ifdef RT_USING_MEMTRACE
245  rt_mem_setname(mem, "INIT");
246 #endif
247 
248  /* initialize the end of the heap */
249  heap_end = (struct heap_mem *)&heap_ptr[mem->next];
250  heap_end->magic = HEAP_MAGIC;
251  heap_end->used = 1;
252  heap_end->next = mem_size_aligned + SIZEOF_STRUCT_MEM;
253  heap_end->prev = mem_size_aligned + SIZEOF_STRUCT_MEM;
254 #ifdef RT_USING_MEMTRACE
255  rt_mem_setname(heap_end, "INIT");
256 #endif
257 
258  rt_sem_init(&heap_sem, "heap", 1, RT_IPC_FLAG_FIFO);
259 
260  /* initialize the lowest-free pointer to the start of the heap */
261  lfree = (struct heap_mem *)heap_ptr;
262 }

270 /**
271  * @brief 分配内存块
272  *
273  * 该函数将从内存堆上分配用户指定大小的内存块。
274  *
275  * @param size 需要分配的内存块的大小，单位为字节。
276  *
277  * @return 成功则返回分配的内存块地址；失败则返回 失败
278  */
279 void *rt_malloc(rt_size_t size)
280 {
281  rt_size_t ptr, ptr2;
282  struct heap_mem *mem, *mem2;
283 
284  RT_DEBUG_NOT_IN_INTERRUPT;
285 
286  if (size == 0)
287  return RT_NULL;
288 
289  if (size != RT_ALIGN(size, RT_ALIGN_SIZE))
290  RT_DEBUG_LOG(RT_DEBUG_MEM, ("malloc size %d, but align to %d\n",
291  size, RT_ALIGN(size, RT_ALIGN_SIZE)));
292  else
293  RT_DEBUG_LOG(RT_DEBUG_MEM, ("malloc size %d\n", size));
294 
295  /* alignment size */
296  size = RT_ALIGN(size, RT_ALIGN_SIZE);
297 
298  if (size > mem_size_aligned)
299  {
300  RT_DEBUG_LOG(RT_DEBUG_MEM, ("no memory\n"));
301 
302  return RT_NULL;
303  }
304 
305  /* every data block must be at least MIN_SIZE_ALIGNED long */
306  if (size < MIN_SIZE_ALIGNED)
307  size = MIN_SIZE_ALIGNED;
308 
309  /* take memory semaphore */
310  rt_sem_take(&heap_sem, RT_WAITING_FOREVER);
311 
312  for (ptr = (rt_uint8_t *)lfree - heap_ptr;
313  ptr < mem_size_aligned - size;
314  ptr = ((struct heap_mem *)&heap_ptr[ptr])->next)
315  {
316  mem = (struct heap_mem *)&heap_ptr[ptr];
317 
318  if ((!mem->used) && (mem->next - (ptr + SIZEOF_STRUCT_MEM)) >= size)
319  {
320  /* mem is not used and at least perfect fit is possible:
321  * mem->next - (ptr + SIZEOF_STRUCT_MEM) gives us the 'user data size' of mem */
322 
323  if (mem->next - (ptr + SIZEOF_STRUCT_MEM) >=
324  (size + SIZEOF_STRUCT_MEM + MIN_SIZE_ALIGNED))
325  {
326  /* (in addition to the above, we test if another struct heap_mem (SIZEOF_STRUCT_MEM) containing
327  * at least MIN_SIZE_ALIGNED of data also fits in the 'user data space' of 'mem')
328  * -> split large block, create empty remainder,
329  * remainder must be large enough to contain MIN_SIZE_ALIGNED data: if
330  * mem->next - (ptr + (2*SIZEOF_STRUCT_MEM)) == size,
331  * struct heap_mem would fit in but no data between mem2 and mem2->next
332  * @todo we could leave out MIN_SIZE_ALIGNED. We would create an empty
333  * region that couldn't hold data, but when mem->next gets freed,
334  * the 2 regions would be combined, resulting in more free memory
335  */
336  ptr2 = ptr + SIZEOF_STRUCT_MEM + size;
337 
338  /* create mem2 struct */
339  mem2 = (struct heap_mem *)&heap_ptr[ptr2];
340  mem2->magic = HEAP_MAGIC;
341  mem2->used = 0;
342  mem2->next = mem->next;
343  mem2->prev = ptr;
344 #ifdef RT_USING_MEMTRACE
345  rt_mem_setname(mem2, " ");
346 #endif
347 
348  /* and insert it between mem and mem->next */
349  mem->next = ptr2;
350  mem->used = 1;
351 
352  if (mem2->next != mem_size_aligned + SIZEOF_STRUCT_MEM)
353  {
354  ((struct heap_mem *)&heap_ptr[mem2->next])->prev = ptr2;
355  }
356 #ifdef RT_MEM_STATS
357  used_mem += (size + SIZEOF_STRUCT_MEM);
358  if (max_mem < used_mem)
359  max_mem = used_mem;
360 #endif
361  }
362  else
363  {
364  /* (a mem2 struct does no fit into the user data space of mem and mem->next will always
365  * be used at this point: if not we have 2 unused structs in a row, plug_holes should have
366  * take care of this).
367  * -> near fit or excact fit: do not split, no mem2 creation
368  * also can't move mem->next directly behind mem, since mem->next
369  * will always be used at this point!
370  */
371  mem->used = 1;
372 #ifdef RT_MEM_STATS
373  used_mem += mem->next - ((rt_uint8_t *)mem - heap_ptr);
374  if (max_mem < used_mem)
375  max_mem = used_mem;
376 #endif
377  }
378  /* set memory block magic */
379  mem->magic = HEAP_MAGIC;
380 #ifdef RT_USING_MEMTRACE
381  if (rt_thread_self())
382  rt_mem_setname(mem, rt_thread_self()->name);
383  else
384  rt_mem_setname(mem, "NONE");
385 #endif
386 
387  if (mem == lfree)
388  {
389  /* Find next free block after mem and update lowest free pointer */
390  while (lfree->used && lfree != heap_end)
391  lfree = (struct heap_mem *)&heap_ptr[lfree->next];
392 
393  RT_ASSERT(((lfree == heap_end) || (!lfree->used)));
394  }
395 
396  rt_sem_release(&heap_sem);
397  RT_ASSERT((rt_uint32_t)mem + SIZEOF_STRUCT_MEM + size <= (rt_uint32_t)heap_end);
398  RT_ASSERT((rt_uint32_t)((rt_uint8_t *)mem + SIZEOF_STRUCT_MEM) % RT_ALIGN_SIZE == 0);
399  RT_ASSERT((((rt_uint32_t)mem) & (RT_ALIGN_SIZE - 1)) == 0);
400 
401  RT_DEBUG_LOG(RT_DEBUG_MEM,
402  ("allocate memory at 0x%x, size: %d\n",
403  (rt_uint32_t)((rt_uint8_t *)mem + SIZEOF_STRUCT_MEM),
404  (rt_uint32_t)(mem->next - ((rt_uint8_t *)mem - heap_ptr))));
405 
406  RT_OBJECT_HOOK_CALL(rt_malloc_hook,
407  (((void *)((rt_uint8_t *)mem + SIZEOF_STRUCT_MEM)), size));
408 
409  /* return the memory data except mem struct */
410  return (rt_uint8_t *)mem + SIZEOF_STRUCT_MEM;
411  }
412  }
413 
414  rt_sem_release(&heap_sem);
415 
416  return RT_NULL;
417 }
418 RTM_EXPORT(rt_malloc);

420 /**
421  * @brief 重新分配内存块
422  *
423  * 此函数将在已分配内存块的基础上重新分配内存块的大小（增加或缩小），在进行重新分配内存块时，
424  * 原来的内存块数据保持不变（缩小的情况下，后面的数据被自动截断）。
425  *
426  * @param rmem 指向已分配的内存块
427  * @param newsize 重新分配的内存大小
428  *
429  * @return 重新分配的内存块地址
430  */
431 void *rt_realloc(void *rmem, rt_size_t newsize)
432 {
433  rt_size_t size;
434  rt_size_t ptr, ptr2;
435  struct heap_mem *mem, *mem2;
436  void *nmem;
437 
438  RT_DEBUG_NOT_IN_INTERRUPT;
439 
440  /* alignment size */
441  newsize = RT_ALIGN(newsize, RT_ALIGN_SIZE);
442  if (newsize > mem_size_aligned)
443  {
444  RT_DEBUG_LOG(RT_DEBUG_MEM, ("realloc: out of memory\n"));
445 
446  return RT_NULL;
447  }
448  else if (newsize == 0)
449  {
450  rt_free(rmem);
451  return RT_NULL;
452  }
453 
454  /* allocate a new memory block */
455  if (rmem == RT_NULL)
456  return rt_malloc(newsize);
457 
458  rt_sem_take(&heap_sem, RT_WAITING_FOREVER);
459 
460  if ((rt_uint8_t *)rmem < (rt_uint8_t *)heap_ptr ||
461  (rt_uint8_t *)rmem >= (rt_uint8_t *)heap_end)
462  {
463  /* illegal memory */
464  rt_sem_release(&heap_sem);
465 
466  return rmem;
467  }
468 
469  mem = (struct heap_mem *)((rt_uint8_t *)rmem - SIZEOF_STRUCT_MEM);
470 
471  ptr = (rt_uint8_t *)mem - heap_ptr;
472  size = mem->next - ptr - SIZEOF_STRUCT_MEM;
473  if (size == newsize)
474  {
475  /* the size is the same as */
476  rt_sem_release(&heap_sem);
477 
478  return rmem;
479  }
480 
481  if (newsize + SIZEOF_STRUCT_MEM + MIN_SIZE < size)
482  {
483  /* split memory block */
484 #ifdef RT_MEM_STATS
485  used_mem -= (size - newsize);
486 #endif
487 
488  ptr2 = ptr + SIZEOF_STRUCT_MEM + newsize;
489  mem2 = (struct heap_mem *)&heap_ptr[ptr2];
490  mem2->magic = HEAP_MAGIC;
491  mem2->used = 0;
492  mem2->next = mem->next;
493  mem2->prev = ptr;
494 #ifdef RT_USING_MEMTRACE
495  rt_mem_setname(mem2, " ");
496 #endif
497  mem->next = ptr2;
498  if (mem2->next != mem_size_aligned + SIZEOF_STRUCT_MEM)
499  {
500  ((struct heap_mem *)&heap_ptr[mem2->next])->prev = ptr2;
501  }
502 
503  plug_holes(mem2);
504 
505  rt_sem_release(&heap_sem);
506 
507  return rmem;
508  }
509  rt_sem_release(&heap_sem);
510 
511  /* expand memory */
512  nmem = rt_malloc(newsize);
513  if (nmem != RT_NULL) /* check memory */
514  {
515  rt_memcpy(nmem, rmem, size < newsize ? size : newsize);
516  rt_free(rmem);
517  }
518 
519  return nmem;
520 }
521 RTM_EXPORT(rt_realloc);

550 /**
551  * @brief 释放内存块
552  *
553  * 用户线程使用完从内存分配器中申请的内存后，必须及时调用该函数释放，否则会造成内存泄漏，
554  * 系统会把待释放的内存还回给堆管理器中。
555  *
556  * @param rmem 待释放的内存块指针
557  */
558 void rt_free(void *rmem)
559 {
560  struct heap_mem *mem;
561 
562  RT_DEBUG_NOT_IN_INTERRUPT;
563 
564  if (rmem == RT_NULL)
565  return;
566  RT_ASSERT((((rt_uint32_t)rmem) & (RT_ALIGN_SIZE - 1)) == 0);
567  RT_ASSERT((rt_uint8_t *)rmem >= (rt_uint8_t *)heap_ptr &&
568  (rt_uint8_t *)rmem < (rt_uint8_t *)heap_end);
569 
570  RT_OBJECT_HOOK_CALL(rt_free_hook, (rmem));
571 
572  if ((rt_uint8_t *)rmem < (rt_uint8_t *)heap_ptr ||
573  (rt_uint8_t *)rmem >= (rt_uint8_t *)heap_end)
574  {
575  RT_DEBUG_LOG(RT_DEBUG_MEM, ("illegal memory\n"));
576 
577  return;
578  }
579 
580  /* Get the corresponding struct heap_mem ... */
581  mem = (struct heap_mem *)((rt_uint8_t *)rmem - SIZEOF_STRUCT_MEM);
582 
583  RT_DEBUG_LOG(RT_DEBUG_MEM,
584  ("release memory 0x%x, size: %d\n",
585  (rt_uint32_t)rmem,
586  (rt_uint32_t)(mem->next - ((rt_uint8_t *)mem - heap_ptr))));
587 
588 
589  /* protect the heap from concurrent access */
590  rt_sem_take(&heap_sem, RT_WAITING_FOREVER);
591 
592  /* ... which has to be in a used state ... */
593  if (!mem->used || mem->magic != HEAP_MAGIC)
594  {
595  rt_kprintf("to free a bad data block:\n");
596  rt_kprintf("mem: 0x%08x, used flag: %d, magic code: 0x%04x\n", mem, mem->used, mem->magic);
597  }
598  RT_ASSERT(mem->used);
599  RT_ASSERT(mem->magic == HEAP_MAGIC);
600  /* ... and is now unused. */
601  mem->used = 0;
602  mem->magic = HEAP_MAGIC;
603 #ifdef RT_USING_MEMTRACE
604  rt_mem_setname(mem, " ");
605 #endif
606 
607  if (mem < lfree)
608  {
609  /* the newly freed struct is now the lowest */
610  lfree = mem;
611  }
612 
613 #ifdef RT_MEM_STATS
614  used_mem -= (mem->next - ((rt_uint8_t *)mem - heap_ptr));
615 #endif
616 
617  /* finally, see if prev or next are free also */
618  plug_holes(mem);
619  rt_sem_release(&heap_sem);
620 }
621 RTM_EXPORT(rt_free);

623 #ifdef RT_MEM_STATS
624 void rt_memory_info(rt_uint32_t *total,
625  rt_uint32_t *used,
626  rt_uint32_t *max_used)
627 {
628  if (total != RT_NULL)
