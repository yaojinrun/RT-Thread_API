77 /**
78  * @brief 初始化内存池
79  *
80  * 该函数将初始化一个内存池对象，通常用于静态对象。
81  *
82  * @param mp 内存池对象句柄
83  * @param name 内存池的名称
84  * @param start 内存池的起始地址
85  * @param size 内存池数据区域大小
86  * @param block_size 内存块容量
87  *
88  * @return RT_EOK 初始化成功；- RT_ERROR 失败
89  */
90 rt_err_t rt_mp_init(struct rt_mempool *mp,
91  const char *name,
92  void *start,
93  rt_size_t size,
94  rt_size_t block_size)
95 {
96  rt_uint8_t *block_ptr;
97  register rt_size_t offset;
98 
99  /* parameter check */
100  RT_ASSERT(mp != RT_NULL);
101 
102  /* initialize object */
103  rt_object_init(&(mp->parent), RT_Object_Class_MemPool, name);
104 
105  /* initialize memory pool */
106  mp->start_address = start;
107  mp->size = RT_ALIGN_DOWN(size, RT_ALIGN_SIZE);
108 
109  /* align the block size */
110  block_size = RT_ALIGN(block_size, RT_ALIGN_SIZE);
111  mp->block_size = block_size;
112 
113  /* align to align size byte */
114  mp->block_total_count = mp->size / (mp->block_size + sizeof(rt_uint8_t *));
115  mp->block_free_count = mp->block_total_count;
116 
117  /* initialize suspended thread list */
118  rt_list_init(&(mp->suspend_thread));
119  mp->suspend_thread_count = 0;
120 
121  /* initialize free block list */
122  block_ptr = (rt_uint8_t *)mp->start_address;
123  for (offset = 0; offset < mp->block_total_count; offset ++)
124  {
125  *(rt_uint8_t **)(block_ptr + offset * (block_size + sizeof(rt_uint8_t *))) =
126  (rt_uint8_t *)(block_ptr + (offset + 1) * (block_size + sizeof(rt_uint8_t *)));
127  }
128 
129  *(rt_uint8_t **)(block_ptr + (offset - 1) * (block_size + sizeof(rt_uint8_t *))) =
130  RT_NULL;
131 
132  mp->block_list = block_ptr;
133 
134  return RT_EOK;
135 }

138 /**
139  * @brief 脱离内存池
140  *
141  * 该函数将把内存池对象从内核对象管理器中删除。使用该函数接口后，
142  * 内核先唤醒所有等待在该内存池对象上的线程，然后将内存池对象从
143  * 内核对象管理器中删除。
144  *
145  * @param mp 内存池对象
146  *
147  * @return RT_EOK
148  */
149 rt_err_t rt_mp_detach(struct rt_mempool *mp)
150 {
151  struct rt_thread *thread;
152  register rt_ubase_t temp;
153 
154  /* parameter check */
155  RT_ASSERT(mp != RT_NULL);
156 
157  /* wake up all suspended threads */
158  while (!rt_list_isempty(&(mp->suspend_thread)))
159  {
160  /* disable interrupt */
161  temp = rt_hw_interrupt_disable();
162 
163  /* get next suspend thread */
164  thread = rt_list_entry(mp->suspend_thread.next, struct rt_thread, tlist);
165  /* set error code to RT_ERROR */
166  thread->error = -RT_ERROR;
167 
168  /*
169  * resume thread
170  * In rt_thread_resume function, it will remove current thread from
171  * suspend list
172  */
173  rt_thread_resume(thread);
174 
175  /* decrease suspended thread count */
176  mp->suspend_thread_count --;
177 
178  /* enable interrupt */
179  rt_hw_interrupt_enable(temp);
180  }
181 
182  /* detach object */
183  rt_object_detach(&(mp->parent));
184 
185  return RT_EOK;
186 }

315 /**
316  * @brief 分配内存块
317  *
318  * 该函数将从指定的内存池中分配一个内存块，如果内存池中有可用的内存块，则从内存池的
319  * 空闲块链表上取下一个内存块，减少空闲块数目并返回这个内存块；如果内存池中已经没有
320  * 空闲内存块，则判断超时时间设置：若超时时间设置为零，则立刻返回空内存块；若等待
321  * 时间大于零，则把当前线程挂起在该内存池对象上，直到内存池中有可用的自由内存块，
322  * 或等待时间到达。
323  *
324  * @param mp 内存池对象
325  * @param time 超时时间
326  *
327  * @return 成功则返回分配的内存块，失败则返回 RT_NULL
328  */
329 void *rt_mp_alloc(rt_mp_t mp, rt_int32_t time)
330 {
331  rt_uint8_t *block_ptr;
332  register rt_base_t level;
333  struct rt_thread *thread;
334  rt_uint32_t before_sleep = 0;
335 
336  /* get current thread */
337  thread = rt_thread_self();
338 
339  /* disable interrupt */
340  level = rt_hw_interrupt_disable();
341 
342  while (mp->block_free_count == 0)
343  {
344  /* memory block is unavailable. */
345  if (time == 0)
346  {
347  /* enable interrupt */
348  rt_hw_interrupt_enable(level);
349 
350  rt_set_errno(-RT_ETIMEOUT);
351 
352  return RT_NULL;
353  }
354 
355  RT_DEBUG_NOT_IN_INTERRUPT;
356 
357  thread->error = RT_EOK;
358 
359  /* need suspend thread */
360  rt_thread_suspend(thread);
361  rt_list_insert_after(&(mp->suspend_thread), &(thread->tlist));
362  mp->suspend_thread_count++;
363 
364  if (time > 0)
365  {
366  /* get the start tick of timer */
367  before_sleep = rt_tick_get();
368 
369  /* init thread timer and start it */
370  rt_timer_control(&(thread->thread_timer),
371  RT_TIMER_CTRL_SET_TIME,
372  &time);
373  rt_timer_start(&(thread->thread_timer));
374  }
375 
376  /* enable interrupt */
377  rt_hw_interrupt_enable(level);
378 
379  /* do a schedule */
380  rt_schedule();
381 
382  if (thread->error != RT_EOK)
383  return RT_NULL;
384 
385  if (time > 0)
386  {
387  time -= rt_tick_get() - before_sleep;
388  if (time < 0)
389  time = 0;
390  }
391  /* disable interrupt */
392  level = rt_hw_interrupt_disable();
393  }
394 
395  /* memory block is available. decrease the free block counter */
396  mp->block_free_count--;
397 
398  /* get block from block list */
399  block_ptr = mp->block_list;
400  RT_ASSERT(block_ptr != RT_NULL);
401 
402  /* Setup the next free node. */
403  mp->block_list = *(rt_uint8_t **)block_ptr;
404 
405  /* point to memory pool */
406  *(rt_uint8_t **)block_ptr = (rt_uint8_t *)mp;
407 
408  /* enable interrupt */
409  rt_hw_interrupt_enable(level);
410 
411  RT_OBJECT_HOOK_CALL(rt_mp_alloc_hook,
412  (mp, (rt_uint8_t *)(block_ptr + sizeof(rt_uint8_t *))));
413 
414  return (rt_uint8_t *)(block_ptr + sizeof(rt_uint8_t *));
415 }
416 RTM_EXPORT(rt_mp_alloc);

418 /**
419  * @brief 释放内存块
420  *
421  * 任何内存块使用完后都必须被释放，否则会造成内存泄露。调用该函数释放指定的内存块，
422  * 首先通过需要被释放的内存块指针计算出该内存块所在的（或所属于的）内存池对象，
423  * 然后增加内存池对象的可用内存块数目，并把该被释放的内存块加入空闲内存块链表上。
424  * 接着判断该内存池对象上是否有挂起的线程，如果有，则唤醒挂起线程链表上的首线程。
425  *
426  * @param block 将要被释放的内存块指针
427  */
428 void rt_mp_free(void *block)
429 {
430  rt_uint8_t **block_ptr;
431  struct rt_mempool *mp;
432  struct rt_thread *thread;
433  register rt_base_t level;
434 
435  /* get the control block of pool which the block belongs to */
436  block_ptr = (rt_uint8_t **)((rt_uint8_t *)block - sizeof(rt_uint8_t *));
437  mp = (struct rt_mempool *)*block_ptr;
438 
439  RT_OBJECT_HOOK_CALL(rt_mp_free_hook, (mp, block));
440 
441  /* disable interrupt */
442  level = rt_hw_interrupt_disable();
443 
444  /* increase the free block count */
445  mp->block_free_count ++;
446 
447  /* link the block into the block list */
448  *block_ptr = mp->block_list;
449  mp->block_list = (rt_uint8_t *)block_ptr;
450 
451  if (mp->suspend_thread_count > 0)
452  {
453  /* get the suspended thread */
454  thread = rt_list_entry(mp->suspend_thread.next,
455  struct rt_thread,
456  tlist);
457 
458  /* set error */
459  thread->error = RT_EOK;
460 
461  /* resume thread */
462  rt_thread_resume(thread);
463 
464  /* decrease suspended thread count */
465  mp->suspend_thread_count --;
466 
467  /* enable interrupt */
468  rt_hw_interrupt_enable(level);
469 
470  /* do a schedule */
471  rt_schedule();
472 
473  return;
474  }
475 
476  /* enable interrupt */
477  rt_hw_interrupt_enable(level);
478 }
479 RTM_EXPORT(rt_mp_free);
