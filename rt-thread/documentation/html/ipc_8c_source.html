83 /**
84  * @brief 将线程挂起到对象链表
85  *
86  * 该函数将挂起一个线程到指定的链表。 IPC对象或某些双队列对象（邮箱等）包含这种链表。
87  *
88  * @param list 该IPC上挂起的线程链表
89  * @param thread 要刮起的线程对象
90  * @param flag IPC对象的标志位，应为RT_IPC_FLAG_FIFO / RT_IPC_FLAG_PRIO。
91  *
92  * @return RT_EOK
93  */
94 rt_inline rt_err_t rt_ipc_list_suspend(rt_list_t *list,
95  struct rt_thread *thread,
96  rt_uint8_t flag)
97 {
98  /* suspend thread */
99  rt_thread_suspend(thread);
100 
101  switch (flag)
102  {
103  case RT_IPC_FLAG_FIFO:
104  rt_list_insert_before(list, &(thread->tlist));
105  break;
106 
107  case RT_IPC_FLAG_PRIO:
108  {
109  struct rt_list_node *n;
110  struct rt_thread *sthread;
111 
112  /* find a suitable position */
113  for (n = list->next; n != list; n = n->next)
114  {
115  sthread = rt_list_entry(n, struct rt_thread, tlist);
116 
117  /* find out */
118  if (thread->current_priority < sthread->current_priority)
119  {
120  /* insert this thread before the sthread */
121  rt_list_insert_before(&(sthread->tlist), &(thread->tlist));
122  break;
123  }
124  }
125 
126  /*
127  * not found a suitable position,
128  * append to the end of suspend_thread list
129  */
130  if (n == list)
131  rt_list_insert_before(list, &(thread->tlist));
132  }
133  break;
134  }
135 
136  return RT_EOK;
137 }

166  * @brief 恢复IPC对象上挂起的所有线程
167  *
168  *此函数将恢复列表中所有挂起的线程，包括IPC对象的挂起链表和邮箱的私有链表等。
169  *
170  * @param list 将要恢复的线程链表
171  *
172  * @return RT_EOK

174 rt_inline rt_err_t rt_ipc_list_resume_all(rt_list_t *list)
175 {
176  struct rt_thread *thread;
177  register rt_ubase_t temp;
178 
179  /* wakeup all suspend threads */
180  while (!rt_list_isempty(list))
181  {
182  /* disable interrupt */
183  temp = rt_hw_interrupt_disable();
184 
185  /* get next suspend thread */
186  thread = rt_list_entry(list->next, struct rt_thread, tlist);
187  /* set error code to RT_ERROR */
188  thread->error = -RT_ERROR;
189 
190  /*
191  * resume thread
192  * In rt_thread_resume function, it will remove current thread from
193  * suspend list
194  */
195  rt_thread_resume(thread);
196 
197  /* enable interrupt */
198  rt_hw_interrupt_enable(temp);
199  }
200 
201  return RT_EOK;
202 }

341 /**
342  * @brief 获取信号量
343  *
344  * 线程通过调用该函数获取信号量来获得信号量资源实例，当信号量值大于零时，线程将获得信号量，
345  * 并且相应的信号量值会减 1；如果信号量的值等于零，那么说明当前信号量资源实例不可用，申请
346  * 该信号量的线程将根据 time 参数的情况选择直接返回、或挂起等待一段时间、或永久等待，直到
347  * 其他线程或中断释放该信号量。
348  *
349  * @param sem 信号量对象的句柄
350  * @param time 指定的等待时间，单位是操作系统时钟节拍（OS Tick）
351  *
352  * @return RT_EOK 成功获得信号量；-RT_ETIMEOUT 超时依然未获得信号量；-RT_ERROR 其他错误
353  */
354 rt_err_t rt_sem_take(rt_sem_t sem, rt_int32_t time)
355 {
356  register rt_base_t temp;
357  struct rt_thread *thread;
358 
359  RT_ASSERT(sem != RT_NULL);
360 
361  RT_OBJECT_HOOK_CALL(rt_object_trytake_hook, (&(sem->parent.parent)));
362 
363  /* disable interrupt */
364  temp = rt_hw_interrupt_disable();
365 
366  RT_DEBUG_LOG(RT_DEBUG_IPC, ("thread %s take sem:%s, which value is: %d\n",
367  rt_thread_self()->name,
368  ((struct rt_object *)sem)->name,
369  sem->value));
370 
371  if (sem->value > 0)
372  {
373  /* semaphore is available */
374  sem->value --;
375 
376  /* enable interrupt */
377  rt_hw_interrupt_enable(temp);
378  }
379  else
380  {
381  /* no waiting, return with timeout */
382  if (time == 0)
383  {
384  rt_hw_interrupt_enable(temp);
385 
386  return -RT_ETIMEOUT;
387  }
388  else
389  {
390  /* current context checking */
391  RT_DEBUG_IN_THREAD_CONTEXT;
392 
393  /* semaphore is unavailable, push to suspend list */
394  /* get current thread */
395  thread = rt_thread_self();
396 
397  /* reset thread error number */
398  thread->error = RT_EOK;
399 
400  RT_DEBUG_LOG(RT_DEBUG_IPC, ("sem take: suspend thread - %s\n",
401  thread->name));
402 
403  /* suspend thread */
404  rt_ipc_list_suspend(&(sem->parent.suspend_thread),
405  thread,
406  sem->parent.parent.flag);
407 
408  /* has waiting time, start thread timer */
409  if (time > 0)
410  {
411  RT_DEBUG_LOG(RT_DEBUG_IPC, ("set thread:%s to timer list\n",
412  thread->name));
413 
414  /* reset the timeout of thread timer and start it */
415  rt_timer_control(&(thread->thread_timer),
416  RT_TIMER_CTRL_SET_TIME,
417  &time);
418  rt_timer_start(&(thread->thread_timer));
419  }
420 
421  /* enable interrupt */
422  rt_hw_interrupt_enable(temp);
423 
424  /* do schedule */
425  rt_schedule();
426 
427  if (thread->error != RT_EOK)
428  {
429  return thread->error;
430  }
431  }
432  }
433 
434  RT_OBJECT_HOOK_CALL(rt_object_take_hook, (&(sem->parent.parent)));
435 
436  return RT_EOK;
437 }
438 RTM_EXPORT(rt_sem_take);

455 /**
456  * @brief 释放信号量
457  *
458  * 该函数将释放一个信号量，当信号量的值等于零时，并且有线程等待这个信号量时，
459  * 释放信号量将唤醒等待在该信号量线程队列中的第一个线程，由它获取信号量；否则
460  * 将把信号量的值加一。
461  *
462  * @param sem 信号量对象的句柄
463  *
464  * @return RT_EOK 成功释放信号量
465  */
466 rt_err_t rt_sem_release(rt_sem_t sem)
467 {
468  register rt_base_t temp;
469  register rt_bool_t need_schedule;
470 
471  RT_OBJECT_HOOK_CALL(rt_object_put_hook, (&(sem->parent.parent)));
472 
473  need_schedule = RT_FALSE;
474 
475  /* disable interrupt */
476  temp = rt_hw_interrupt_disable();
477 
478  RT_DEBUG_LOG(RT_DEBUG_IPC, ("thread %s releases sem:%s, which value is: %d\n",
479  rt_thread_self()->name,
480  ((struct rt_object *)sem)->name,
481  sem->value));
482 
483  if (!rt_list_isempty(&sem->parent.suspend_thread))
484  {
485  /* resume the suspended thread */
486  rt_ipc_list_resume(&(sem->parent.suspend_thread));
487  need_schedule = RT_TRUE;
488  }
489  else
490  sem->value ++; /* increase value */
491 
492  /* enable interrupt */
493  rt_hw_interrupt_enable(temp);
494 
495  /* resume a thread, re-schedule */
496  if (need_schedule == RT_TRUE)
497  rt_schedule();
498 
499  return RT_EOK;
500 }
501 RTM_EXPORT(rt_sem_release);

682  * @brief 获取互斥量
683  *
684  * 该函数将获取一个互斥量，如果互斥量没有被其他线程控制，那么申请该互斥量的线程将成功获得该互斥量。
685  * 如果互斥量已经被当前线程线程控制，则该互斥量的持有计数加 1，当前线程也不会挂起等待。如果互斥量
686  * 已经被其他线程占有，则当前线程在该互斥量上挂起等待，直到其他线程释放它或者等待时间超过指定的
687  * 超时时间。
688  *
689  * @param mutex 互斥量对象的句柄
690  * @param time 指定的等待时间
691  *
692  * @return RT_EOK 成功获得互斥量；-RT_ETIMEOUT 超时；-RT_ERROR 获取失败
693  */
694 rt_err_t rt_mutex_take(rt_mutex_t mutex, rt_int32_t time)
695 {
696  register rt_base_t temp;
697  struct rt_thread *thread;
698 
699  /* this function must not be used in interrupt even if time = 0 */
700  RT_DEBUG_IN_THREAD_CONTEXT;
701 
702  RT_ASSERT(mutex != RT_NULL);
703 
704  /* get current thread */
705  thread = rt_thread_self();
706 
707  /* disable interrupt */
708  temp = rt_hw_interrupt_disable();
709 
710  RT_OBJECT_HOOK_CALL(rt_object_trytake_hook, (&(mutex->parent.parent)));
711 
712  RT_DEBUG_LOG(RT_DEBUG_IPC,
713  ("mutex_take: current thread %s, mutex value: %d, hold: %d\n",
714  thread->name, mutex->value, mutex->hold));
715 
716  /* reset thread error */
717  thread->error = RT_EOK;
718 
719  if (mutex->owner == thread)
720  {
721  /* it's the same thread */
722  mutex->hold ++;
723  }
724  else
725  {
726 __again:
727  /* The value of mutex is 1 in initial status. Therefore, if the
728  * value is great than 0, it indicates the mutex is avaible.
729  */
730  if (mutex->value > 0)
731  {
732  /* mutex is available */
733  mutex->value --;
734 
735  /* set mutex owner and original priority */
736  mutex->owner = thread;
737  mutex->original_priority = thread->current_priority;
738  mutex->hold ++;
739  }
740  else
741  {
742  /* no waiting, return with timeout */
743  if (time == 0)
744  {
745  /* set error as timeout */
746  thread->error = -RT_ETIMEOUT;
747 
748  /* enable interrupt */
749  rt_hw_interrupt_enable(temp);
750 
751  return -RT_ETIMEOUT;
752  }
753  else
754  {
755  /* mutex is unavailable, push to suspend list */
756  RT_DEBUG_LOG(RT_DEBUG_IPC, ("mutex_take: suspend thread: %s\n",
757  thread->name));
758 
759  /* change the owner thread priority of mutex */
760  if (thread->current_priority < mutex->owner->current_priority)
761  {
762  /* change the owner thread priority */
763  rt_thread_control(mutex->owner,
764  RT_THREAD_CTRL_CHANGE_PRIORITY,
765  &thread->current_priority);
766  }
767 
768  /* suspend current thread */
769  rt_ipc_list_suspend(&(mutex->parent.suspend_thread),
770  thread,
771  mutex->parent.parent.flag);
772 
773  /* has waiting time, start thread timer */
774  if (time > 0)
775  {
776  RT_DEBUG_LOG(RT_DEBUG_IPC,
777  ("mutex_take: start the timer of thread:%s\n",
778  thread->name));
779 
780  /* reset the timeout of thread timer and start it */
781  rt_timer_control(&(thread->thread_timer),
782  RT_TIMER_CTRL_SET_TIME,
783  &time);
784  rt_timer_start(&(thread->thread_timer));
785  }
786 
787  /* enable interrupt */
788  rt_hw_interrupt_enable(temp);
789 
790  /* do schedule */
791  rt_schedule();
792 
793  if (thread->error != RT_EOK)
794  {
795  /* interrupt by signal, try it again */
796  if (thread->error == -RT_EINTR) goto __again;
797 
798  /* return error */
799  return thread->error;
800  }
801  else
802  {
803  /* the mutex is taken successfully. */
804  /* disable interrupt */
805  temp = rt_hw_interrupt_disable();
806  }
807  }
808  }
809  }
810 
811  /* enable interrupt */
812  rt_hw_interrupt_enable(temp);
813 
814  RT_OBJECT_HOOK_CALL(rt_object_take_hook, (&(mutex->parent.parent)));
815 
816  return RT_EOK;
817 }

821  * @brief 释放互斥量
822  *
823  * 使用该函数接口时，只有已经拥有互斥量控制权的线程才能释放它，每释放一次该互斥量，
824  * 它的持有计数就减 1。当该互斥量的持有计数为零时（即持有线程已经释放所有的持有操作），
825  * 它变为可用，等待在该信号量上的线程将被唤醒。如果线程的运行优先级被互斥量提升，
826  * 那么当互斥量被释放后，线程恢复为持有互斥量前的优先级。
827  *
828  * @param mutex 互斥量对象的句柄
829  *
830  * @return RT_EOK 成功
831  */
832 rt_err_t rt_mutex_release(rt_mutex_t mutex)
833 {
834  register rt_base_t temp;
835  struct rt_thread *thread;
836  rt_bool_t need_schedule;
837 
838  need_schedule = RT_FALSE;
839 
840  /* only thread could release mutex because we need test the ownership */
841  RT_DEBUG_IN_THREAD_CONTEXT;
842 
843  /* get current thread */
844  thread = rt_thread_self();
845 
846  /* disable interrupt */
847  temp = rt_hw_interrupt_disable();
848 
849  RT_DEBUG_LOG(RT_DEBUG_IPC,
850  ("mutex_release:current thread %s, mutex value: %d, hold: %d\n",
851  thread->name, mutex->value, mutex->hold));
852 
853  RT_OBJECT_HOOK_CALL(rt_object_put_hook, (&(mutex->parent.parent)));
854 
855  /* mutex only can be released by owner */
856  if (thread != mutex->owner)
857  {
858  thread->error = -RT_ERROR;
859 
860  /* enable interrupt */
861  rt_hw_interrupt_enable(temp);
862 
863  return -RT_ERROR;
864  }
865 
866  /* decrease hold */
867  mutex->hold --;
868  /* if no hold */
869  if (mutex->hold == 0)
870  {
871  /* change the owner thread to original priority */
872  if (mutex->original_priority != mutex->owner->current_priority)
873  {
874  rt_thread_control(mutex->owner,
875  RT_THREAD_CTRL_CHANGE_PRIORITY,
876  &(mutex->original_priority));
877  }
878 
879  /* wakeup suspended thread */
880  if (!rt_list_isempty(&mutex->parent.suspend_thread))
881  {
882  /* get suspended thread */
883  thread = rt_list_entry(mutex->parent.suspend_thread.next,
884  struct rt_thread,
885  tlist);
886 
887  RT_DEBUG_LOG(RT_DEBUG_IPC, ("mutex_release: resume thread: %s\n",
888  thread->name));
889 
890  /* set new owner and priority */
891  mutex->owner = thread;
892  mutex->original_priority = thread->current_priority;
893  mutex->hold ++;
894 
895  /* resume thread */
896  rt_ipc_list_resume(&(mutex->parent.suspend_thread));
897 
898  need_schedule = RT_TRUE;
899  }
900  else
901  {
902  /* increase value */
903  mutex->value ++;
904 
905  /* clear owner */
906  mutex->owner = RT_NULL;
907  mutex->original_priority = 0xff;
908  }
909  }
910 
911  /* enable interrupt */
912  rt_hw_interrupt_enable(temp);
913 
914  /* perform a schedule */
915  if (need_schedule == RT_TRUE)
916  rt_schedule();
917 
918  return RT_EOK;
919 }

1085  register rt_base_t status;

1091  return -RT_ERROR;

1102

1104  {

1116  {

1118  status = RT_EOK;
1119  }
1120  }
1121  else if (thread->event_info & RT_EVENT_FLAG_OR)
1122  {

1124  {
1125  /* save recieved event set */
1126  thread->event_set = thread->event_set & event->set;
1127 
1128  /* received an OR event */
1129  status = RT_EOK;
1130  }
1131  }
