101 void rt_thread_exit(void)
102 {
103  struct rt_thread *thread;
104  register rt_base_t level;
105 
106  /* get current thread */
107  thread = rt_current_thread;
108 
109  /* disable interrupt */
110  level = rt_hw_interrupt_disable();
111 
112  /* remove from schedule */
113  rt_schedule_remove_thread(thread);
114  /* change stat */
115  thread->stat = RT_THREAD_CLOSE;
116 
117  /* remove it from timer list */
118  rt_timer_detach(&thread->thread_timer);
119 
120  if ((rt_object_is_systemobject((rt_object_t)thread) == RT_TRUE) &&
121  thread->cleanup == RT_NULL)
122  {
123  rt_object_detach((rt_object_t)thread);
124  }
125  else
126  {
127  /* insert to defunct thread list */
128  rt_list_insert_after(&rt_thread_defunct, &(thread->tlist));
129  }
130 
131  /* enable interrupt */
132  rt_hw_interrupt_enable(level);
133 
134  /* switch to next task */
135  rt_schedule();
136 }

138 static rt_err_t _rt_thread_init(struct rt_thread *thread,
139  const char *name,
140  void (*entry)(void *parameter),
141  void *parameter,
142  void *stack_start,
143  rt_uint32_t stack_size,
144  rt_uint8_t priority,
145  rt_uint32_t tick)
146 {
147  /* init thread list */
148  rt_list_init(&(thread->tlist));
149 
150  thread->entry = (void *)entry;
151  thread->parameter = parameter;
152 
153  /* stack init */
154  thread->stack_addr = stack_start;
155  thread->stack_size = stack_size;
156 
157  /* init thread stack */
158  rt_memset(thread->stack_addr, '#', thread->stack_size);
159  thread->sp = (void *)rt_hw_stack_init(thread->entry, thread->parameter,
160  (void *)((char *)thread->stack_addr + thread->stack_size - 4),
161  (void *)rt_thread_exit);
162 
163  /* priority init */
164  RT_ASSERT(priority < RT_THREAD_PRIORITY_MAX);
165  thread->init_priority = priority;
166  thread->current_priority = priority;
167 
168  thread->number_mask = 0;
169 #if RT_THREAD_PRIORITY_MAX > 32
170  thread->number = 0;
171  thread->high_mask = 0;
172 #endif
173 
174  /* tick init */
175  thread->init_tick = tick;
176  thread->remaining_tick = tick;
177 
178  /* error and flags */
179  thread->error = RT_EOK;
180  thread->stat = RT_THREAD_INIT;
181 
182  /* initialize cleanup function and user data */
183  thread->cleanup = 0;
184  thread->user_data = 0;
185 
186  /* init thread timer */
187  rt_timer_init(&(thread->thread_timer),
188  thread->name,
189  rt_thread_timeout,
190  thread,
191  0,
192  RT_TIMER_FLAG_ONE_SHOT);
193 
194 #ifdef RT_USING_MODULE
195  thread->module_id = RT_NULL;
196 #endif
197 
198  /* initialize signal */
199 #ifdef RT_USING_SIGNALS
200  thread->sig_mask = 0x00;
201  thread->sig_pending = 0x00;
202 
203  thread->sig_ret = RT_NULL;
204  thread->sig_vectors = RT_NULL;
205  thread->si_list = RT_NULL;
206 #endif
207 
208  RT_OBJECT_HOOK_CALL(rt_thread_inited_hook, (thread));
209 
210  return RT_EOK;
211 }

222  * 此函数将初始化一个线程，通常用于初始化一个静态线程对象。
223  *
224  * @param thread 线程句柄。线程句柄由用户提供出来，并指向对应的线程控制块内存地址。

275 /**
276  * @brief 启动线程
277  *
278  * 此函数将启动一个线程并将其放入系统就绪队列
279  *
280  * @param thread 被要被启动的线程句柄
281  *
282  * @return 成功返回 RT_EOK, 失败则返回 -RT_ERROR
283  */
284 rt_err_t rt_thread_startup(rt_thread_t thread)
285 {
286  /* thread check */
287  RT_ASSERT(thread != RT_NULL);
288  RT_ASSERT((thread->stat & RT_THREAD_STAT_MASK) == RT_THREAD_INIT);
289 
290  /* set current priority to init priority */
291  thread->current_priority = thread->init_priority;
292 
293  /* calculate priority attribute */
294 #if RT_THREAD_PRIORITY_MAX > 32
295  thread->number = thread->current_priority >> 3; /* 5bit */
296  thread->number_mask = 1L << thread->number;
297  thread->high_mask = 1L << (thread->current_priority & 0x07); /* 3bit */
298 #else
299  thread->number_mask = 1L << thread->current_priority;
300 #endif
301 
302  RT_DEBUG_LOG(RT_DEBUG_THREAD, ("startup a thread:%s with priority:%d\n",
303  thread->name, thread->init_priority));
304  /* change thread stat */
305  thread->stat = RT_THREAD_SUSPEND;
306  /* then resume it */
307  rt_thread_resume(thread);
308  if (rt_thread_self() != RT_NULL)
309  {
310  /* do a scheduling */
311  rt_schedule();
312  }
313 
314  return RT_EOK;
315 }
316 RTM_EXPORT(rt_thread_startup);

318 /**
319  * @brief 脱离线程
320  *
321  * 该函数将脱离一个线程。 线程对象将从线程队列中删除，并从系统对象管理中脱离/删除。
322  *
323  * @param thread 线程句柄，它应该是由 rt_thread_init 进行初始化的线程句柄。
324  *
325  * @return 成功返回 RT_EOK, 失败则返回 -RT_ERROR
326  */
327 rt_err_t rt_thread_detach(rt_thread_t thread)
328 {
329  rt_base_t lock;
330 
331  /* thread check */
332  RT_ASSERT(thread != RT_NULL);
333 
334  if ((thread->stat & RT_THREAD_STAT_MASK) != RT_THREAD_INIT)
335  {
336  /* remove from schedule */
337  rt_schedule_remove_thread(thread);
338  }
339 
340  /* release thread timer */
341  rt_timer_detach(&(thread->thread_timer));
342 
343  /* change stat */
344  thread->stat = RT_THREAD_CLOSE;
345 
346  /* detach object */
347  rt_object_detach((rt_object_t)thread);
348 
349  if (thread->cleanup != RT_NULL)
350  {
351  /* disable interrupt */
352  lock = rt_hw_interrupt_disable();
353 
354  /* insert to defunct thread list */
355  rt_list_insert_after(&rt_thread_defunct, &(thread->tlist));
356 
357  /* enable interrupt */
358  rt_hw_interrupt_enable(lock);
359  }
360 
361  return RT_EOK;
362 }

460 /**
461  * @brief 使线程让出处理器资源
462  *
463  * 该函数将使当前线程出让处理器资源，它将不再占有处理器，调度器会选择相同优先级的下一个线程执行。
464  * 线程调用这个接口后，这个线程仍然在就绪队列中。
465  *
466  * @return RT_EOK
467  */
468 rt_err_t rt_thread_yield(void)
469 {
470  register rt_base_t level;
471  struct rt_thread *thread;
472 
473  /* disable interrupt */
474  level = rt_hw_interrupt_disable();
475 
476  /* set to current thread */
477  thread = rt_current_thread;
478 
479  /* if the thread stat is READY and on ready queue list */
480  if ((thread->stat & RT_THREAD_STAT_MASK) == RT_THREAD_READY &&
481  thread->tlist.next != thread->tlist.prev)
482  {
483  /* remove thread from thread list */
484  rt_list_remove(&(thread->tlist));
485 
486  /* put thread to end of ready queue */
487  rt_list_insert_before(&(rt_thread_priority_table[thread->current_priority]),
488  &(thread->tlist));
489 
490  /* enable interrupt */
491  rt_hw_interrupt_enable(level);
492 
493  rt_schedule();
494 
495  return RT_EOK;
496  }
497 
498  /* enable interrupt */
499  rt_hw_interrupt_enable(level);
500 
501  return RT_EOK;
502 }

505 /**
506  * @brief 使线程睡眠
507  *
508  * 该函数将使当前线程睡眠几个系统始终节拍的时间
509  *
510  * @param tick 线程睡眠的时间
511  *
512  * @return RT_EOK
513  */
514 rt_err_t rt_thread_sleep(rt_tick_t tick)
515 {
516  register rt_base_t temp;
517  struct rt_thread *thread;
518 
519  /* disable interrupt */
520  temp = rt_hw_interrupt_disable();
521  /* set to current thread */
522  thread = rt_current_thread;
523  RT_ASSERT(thread != RT_NULL);
524 
525  /* suspend thread */
526  rt_thread_suspend(thread);
527 
528  /* reset the timeout of thread timer and start it */
529  rt_timer_control(&(thread->thread_timer), RT_TIMER_CTRL_SET_TIME, &tick);
530  rt_timer_start(&(thread->thread_timer));
531 
532  /* enable interrupt */
533  rt_hw_interrupt_enable(temp);
534 
535  rt_schedule();
536 
537  /* clear error number of this thread to RT_EOK */
538  if (thread->error == -RT_ETIMEOUT)
539  thread->error = RT_EOK;
540 
541  return RT_EOK;
542 }

578 /**
579  * @brief 控制线程
580  *
581  * 该函数将根据命令修改线程的一些配置参数。
582  *
583  * @param thread 被指定控制的线程
584  * @param cmd 控制命令, 包含以下参数：
585  * RT_THREAD_CTRL_CHANGE_PRIORITY 改变线程的优先级
586  * RT_THREAD_CTRL_STARTUP 启动线程
587  * RT_THREAD_CTRL_CLOSE 删除线程
588  * @param arg 控制命令相关的参数
589  *
590  * @return RT_EOK
591  */
592 rt_err_t rt_thread_control(rt_thread_t thread, int cmd, void *arg)
593 {
594  register rt_base_t temp;
595 
596  /* thread check */
597  RT_ASSERT(thread != RT_NULL);
598 
599  switch (cmd)
600  {
601  case RT_THREAD_CTRL_CHANGE_PRIORITY:
602  /* disable interrupt */
603  temp = rt_hw_interrupt_disable();
604 
605  /* for ready thread, change queue */
606  if ((thread->stat & RT_THREAD_STAT_MASK) == RT_THREAD_READY)
607  {
608  /* remove thread from schedule queue first */
609  rt_schedule_remove_thread(thread);
610 
611  /* change thread priority */
612  thread->current_priority = *(rt_uint8_t *)arg;
613 
614  /* recalculate priority attribute */
615 #if RT_THREAD_PRIORITY_MAX > 32
616  thread->number = thread->current_priority >> 3; /* 5bit */
617  thread->number_mask = 1 << thread->number;
618  thread->high_mask = 1 << (thread->current_priority & 0x07); /* 3bit */
619 #else
620  thread->number_mask = 1 << thread->current_priority;
621 #endif
622 
623  /* insert thread to schedule queue again */
624  rt_schedule_insert_thread(thread);
625  }
626  else
627  {
628  thread->current_priority = *(rt_uint8_t *)arg;
629 
630  /* recalculate priority attribute */
631 #if RT_THREAD_PRIORITY_MAX > 32
632  thread->number = thread->current_priority >> 3; /* 5bit */
633  thread->number_mask = 1 << thread->number;
634  thread->high_mask = 1 << (thread->current_priority & 0x07); /* 3bit */
635 #else
636  thread->number_mask = 1 << thread->current_priority;
637 #endif
638  }
639 
640  /* enable interrupt */
641  rt_hw_interrupt_enable(temp);
642  break;
643 
644  case RT_THREAD_CTRL_STARTUP:
645  return rt_thread_startup(thread);
646 
647 #ifdef RT_USING_HEAP
648  case RT_THREAD_CTRL_CLOSE:
649  return rt_thread_delete(thread);
650 #endif
651 
652  default:
653  break;
654  }
655 
656  return RT_EOK;
657 }
