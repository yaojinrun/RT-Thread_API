78 /**
79  * 该函数将通知内核经过了一个系统时钟的时间。通常该函数在时钟中断里调用。
80  */
81 void rt_tick_increase(void)
82 {
83  struct rt_thread *thread;
84 
85  /* increase the global tick */
86  ++ rt_tick;
87 
88  /* check time slice */
89  thread = rt_thread_self();
90 
91  -- thread->remaining_tick;
92  if (thread->remaining_tick == 0)
93  {
94  /* change to initialized tick */
95  thread->remaining_tick = thread->init_tick;
96 
97  /* yield */
98  rt_thread_yield();
99  }
100 
101  /* check timer */
102  rt_timer_check();
103 }

105 /**
106  * @brief 节拍到毫秒的转换
107  *
108  * 该函数把毫秒转换为系统时钟节拍计数值。
109  *
110  * @param ms 指定的时间（毫秒）
111  * - 为负数时将永远等待
112  * - 为0时不等待立即返回
113  * - 最大值为 0x7fffffff
114  *
115  * @return 计算后的系统时钟节拍数
116  */
117 int rt_tick_from_millisecond(rt_int32_t ms)
118 {
119  int tick;
120 
121  if (ms < 0)
122  tick = RT_WAITING_FOREVER;
123  else
124  tick = (RT_TICK_PER_SECOND * ms + 999) / 1000;
125 
126  /* return the calculated tick */
127  return tick;
128 }
129 RTM_EXPORT(rt_tick_from_millisecond);
