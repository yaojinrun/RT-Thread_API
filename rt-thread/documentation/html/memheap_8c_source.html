43 /* dynamic pool magic and mask */
44 #define RT_MEMHEAP_MAGIC 0x1ea01ea0
45 #define RT_MEMHEAP_MASK 0xfffffffe
46 #define RT_MEMHEAP_USED 0x01
47 #define RT_MEMHEAP_FREED 0x00
48 
49 #define RT_MEMHEAP_IS_USED(i) ((i)->magic & RT_MEMHEAP_USED)
50 #define RT_MEMHEAP_MINIALLOC 12
51 
52 #define RT_MEMHEAP_SIZE RT_ALIGN(sizeof(struct rt_memheap_item), RT_ALIGN_SIZE)
53 #define MEMITEM_SIZE(item) ((rt_uint32_t)item->next - (rt_uint32_t)item - RT_MEMHEAP_SIZE)
54 
55 /*
56  * The initialized memory pool will be:
57  * +-----------------------------------+--------------------------+
58  * | whole freed memory block | Used Memory Block Tailer |
59  * +-----------------------------------+--------------------------+
60  *
61  * block_list --> whole freed memory block
62  *
63  * The length of Used Memory Block Tailer is 0,
64  * which is prevents block merging across list
65  */
66 
67 /**
68  * @ingroup MM
69  * @brief 初始化memheap
70  *
71  * 在使用memheap堆内存时，必须要在系统初始化的时候进行堆内存的初始化，
72  * 可以通过调用该函数完成内存堆的初始化。
73  *
74  * @param memheap memheap控制块
75  * @param name 内存堆的名称
76  * @param start_addr 堆内存区域起始地址
77  * @param size 堆内存大小
78  *
79  * @return RT_EOK
80  *
81  * @note 如果有多个不连续的memheap可以多次调用该函数将其初始化并加入memheap_item链表。
82  */
83 rt_err_t rt_memheap_init(struct rt_memheap *memheap,
84  const char *name,
85  void *start_addr,
86  rt_uint32_t size)
87 {
88  struct rt_memheap_item *item;
89 
90  RT_ASSERT(memheap != RT_NULL);
91 
92  /* initialize pool object */
93  rt_object_init(&(memheap->parent), RT_Object_Class_MemHeap, name);
94 
95  memheap->start_addr = start_addr;
96  memheap->pool_size = RT_ALIGN_DOWN(size, RT_ALIGN_SIZE);
97  memheap->available_size = memheap->pool_size - (2 * RT_MEMHEAP_SIZE);
98  memheap->max_used_size = memheap->pool_size - memheap->available_size;
99 
100  /* initialize the free list header */
101  item = &(memheap->free_header);
102  item->magic = RT_MEMHEAP_MAGIC;
103  item->pool_ptr = memheap;
104  item->next = RT_NULL;
105  item->prev = RT_NULL;
106  item->next_free = item;
107  item->prev_free = item;
108 
109  /* set the free list to free list header */
110  memheap->free_list = item;
111 
112  /* initialize the first big memory block */
113  item = (struct rt_memheap_item *)start_addr;
114  item->magic = RT_MEMHEAP_MAGIC;
115  item->pool_ptr = memheap;
116  item->next = RT_NULL;
117  item->prev = RT_NULL;
118  item->next_free = item;
119  item->prev_free = item;
120 
121  item->next = (struct rt_memheap_item *)
122  ((rt_uint8_t *)item + memheap->available_size + RT_MEMHEAP_SIZE);
123  item->prev = item->next;
124 
125  /* block list header */
126  memheap->block_list = item;
127 
128  /* place the big memory block to free list */
129  item->next_free = memheap->free_list->next_free;
130  item->prev_free = memheap->free_list;
131  memheap->free_list->next_free->prev_free = item;
132  memheap->free_list->next_free = item;
133 
134  /* move to the end of memory pool to build a small tailer block,
135  * which prevents block merging
136  */
137  item = item->next;
138  /* it's a used memory block */
139  item->magic = RT_MEMHEAP_MAGIC | RT_MEMHEAP_USED;
140  item->pool_ptr = memheap;
141  item->next = (struct rt_memheap_item *)start_addr;
142  item->prev = (struct rt_memheap_item *)start_addr;
143  /* not in free list */
144  item->next_free = item->prev_free = RT_NULL;
145 
146  /* initialize semaphore lock */
147  rt_sem_init(&(memheap->lock), name, 1, RT_IPC_FLAG_FIFO);
148 
149  RT_DEBUG_LOG(RT_DEBUG_MEMHEAP,
150  ("memory heap: start addr 0x%08x, size %d, free list header 0x%08x\n",
151  start_addr, size, &(memheap->free_header)));
152 
153  return RT_EOK;
154 }

169 void *rt_memheap_alloc(struct rt_memheap *heap, rt_uint32_t size)
170 {
171  rt_err_t result;
172  rt_uint32_t free_size;
173  struct rt_memheap_item *header_ptr;
174 
175  RT_ASSERT(heap != RT_NULL);
176 
177  /* align allocated size */
178  size = RT_ALIGN(size, RT_ALIGN_SIZE);
179  if (size < RT_MEMHEAP_MINIALLOC)
180  size = RT_MEMHEAP_MINIALLOC;
181 
182  RT_DEBUG_LOG(RT_DEBUG_MEMHEAP, ("allocate %d on heap:%8.*s",
183  size, RT_NAME_MAX, heap->parent.name));
184 
185  if (size < heap->available_size)
186  {
187  /* search on free list */
188  free_size = 0;
189 
190  /* lock memheap */
191  result = rt_sem_take(&(heap->lock), RT_WAITING_FOREVER);
192  if (result != RT_EOK)
193  {
194  rt_set_errno(result);
195 
196  return RT_NULL;
197  }
198 
199  /* get the first free memory block */
200  header_ptr = heap->free_list->next_free;
201  while (header_ptr != heap->free_list && free_size < size)
202  {
203  /* get current freed memory block size */
204  free_size = MEMITEM_SIZE(header_ptr);
205  if (free_size < size)
206  {
207  /* move to next free memory block */
208  header_ptr = header_ptr->next_free;
209  }
210  }
211 
212  /* determine if the memory is available. */
213  if (free_size >= size)
214  {
215  /* a block that satisfies the request has been found. */
216 
217  /* determine if the block needs to be split. */
218  if (free_size >= (size + RT_MEMHEAP_SIZE + RT_MEMHEAP_MINIALLOC))
219  {
220  struct rt_memheap_item *new_ptr;
221 
222  /* split the block. */
223  new_ptr = (struct rt_memheap_item *)
224  (((rt_uint8_t *)header_ptr) + size + RT_MEMHEAP_SIZE);
225 
226  RT_DEBUG_LOG(RT_DEBUG_MEMHEAP,
227  ("split: block[0x%08x] nextm[0x%08x] prevm[0x%08x] to new[0x%08x]\n",
228  header_ptr,
229  header_ptr->next,
230  header_ptr->prev,
231  new_ptr));
232 
233  /* mark the new block as a memory block and freed. */
234  new_ptr->magic = RT_MEMHEAP_MAGIC;
235 
236  /* put the pool pointer into the new block. */
237  new_ptr->pool_ptr = heap;
238 
239  /* break down the block list */
240  new_ptr->prev = header_ptr;
241  new_ptr->next = header_ptr->next;
242  header_ptr->next->prev = new_ptr;
243  header_ptr->next = new_ptr;
244 
245  /* remove header ptr from free list */
246  header_ptr->next_free->prev_free = header_ptr->prev_free;
247  header_ptr->prev_free->next_free = header_ptr->next_free;
248  header_ptr->next_free = RT_NULL;
249  header_ptr->prev_free = RT_NULL;
250 
251  /* insert new_ptr to free list */
252  new_ptr->next_free = heap->free_list->next_free;
253  new_ptr->prev_free = heap->free_list;
254  heap->free_list->next_free->prev_free = new_ptr;
255  heap->free_list->next_free = new_ptr;
256  RT_DEBUG_LOG(RT_DEBUG_MEMHEAP, ("new ptr: next_free 0x%08x, prev_free 0x%08x\n",
257  new_ptr->next_free,
258  new_ptr->prev_free));
259 
260  /* decrement the available byte count. */
261  heap->available_size = heap->available_size -
262  size -
263  RT_MEMHEAP_SIZE;
264  if (heap->pool_size - heap->available_size > heap->max_used_size)
265  heap->max_used_size = heap->pool_size - heap->available_size;
266  }
267  else
268  {
269  /* decrement the entire free size from the available bytes count. */
270  heap->available_size = heap->available_size - free_size;
271  if (heap->pool_size - heap->available_size > heap->max_used_size)
272  heap->max_used_size = heap->pool_size - heap->available_size;
273 
274  /* remove header_ptr from free list */
275  RT_DEBUG_LOG(RT_DEBUG_MEMHEAP,
276  ("one block: block[0x%08x], next_free 0x%08x, prev_free 0x%08x\n",
277  header_ptr,
278  header_ptr->next_free,
279  header_ptr->prev_free));
280 
281  header_ptr->next_free->prev_free = header_ptr->prev_free;
282  header_ptr->prev_free->next_free = header_ptr->next_free;
283  header_ptr->next_free = RT_NULL;
284  header_ptr->prev_free = RT_NULL;
285  }
286 
287  /* Mark the allocated block as not available. */
288  header_ptr->magic |= RT_MEMHEAP_USED;
289 
290  /* release lock */
291  rt_sem_release(&(heap->lock));
292 
293  /* Return a memory address to the caller. */
294  RT_DEBUG_LOG(RT_DEBUG_MEMHEAP,
295  ("alloc mem: memory[0x%08x], heap[0x%08x], size: %d\n",
296  (void *)((rt_uint8_t *)header_ptr + RT_MEMHEAP_SIZE),
297  header_ptr,
298  size));
299 
300  return (void *)((rt_uint8_t *)header_ptr + RT_MEMHEAP_SIZE);
301  }
302 
303  /* release lock */
304  rt_sem_release(&(heap->lock));
305  }
306 
307  RT_DEBUG_LOG(RT_DEBUG_MEMHEAP, ("allocate memory: failed\n"));
308 
309  /* Return the completion status. */
310  return RT_NULL;
311 }

314 void *rt_memheap_realloc(struct rt_memheap *heap, void *ptr, rt_size_t newsize)
315 {
316  rt_err_t result;
317  rt_size_t oldsize;
318  struct rt_memheap_item *header_ptr;
319  struct rt_memheap_item *new_ptr;
320 
321  if (newsize == 0)
322  {
323  rt_memheap_free(ptr);
324 
325  return RT_NULL;
326  }
327  /* align allocated size */
328  newsize = RT_ALIGN(newsize, RT_ALIGN_SIZE);
329  if (newsize < RT_MEMHEAP_MINIALLOC)
330  newsize = RT_MEMHEAP_MINIALLOC;
331 
332  if (ptr == RT_NULL)
333  {
334  return rt_memheap_alloc(heap, newsize);
335  }
336 
337  /* get memory block header and get the size of memory block */
338  header_ptr = (struct rt_memheap_item *)
339  ((rt_uint8_t *)ptr - RT_MEMHEAP_SIZE);
340  oldsize = MEMITEM_SIZE(header_ptr);
341  /* re-allocate memory */
342  if (newsize > oldsize)
343  {
344  void *new_ptr;
345  struct rt_memheap_item *next_ptr;
346 
347  /* lock memheap */
348  result = rt_sem_take(&(heap->lock), RT_WAITING_FOREVER);
349  if (result != RT_EOK)
350  {
351  rt_set_errno(result);
352  return RT_NULL;
353  }
354 
355  next_ptr = header_ptr->next;
356 
357  /* header_ptr should not be the tail */
358  RT_ASSERT(next_ptr > header_ptr);
359 
360  /* check whether the following free space is enough to expand */
361  if (!RT_MEMHEAP_IS_USED(next_ptr))
362  {
363  rt_int32_t nextsize;
364 
365  nextsize = MEMITEM_SIZE(next_ptr);
366  RT_ASSERT(next_ptr > 0);
367 
368  /* Here is the ASCII art of the situation that we can make use of
369  * the next free node without alloc/memcpy, |*| is the control
370  * block:
371  *
372  * oldsize free node
373  * |*|-----------|*|----------------------|*|
374  * newsize >= minialloc
375  * |*|----------------|*|-----------------|*|
376  */
377  if (nextsize + oldsize > newsize + RT_MEMHEAP_MINIALLOC)
378  {
379  /* decrement the entire free size from the available bytes count. */
380  heap->available_size = heap->available_size - (newsize - oldsize);
381  if (heap->pool_size - heap->available_size > heap->max_used_size)
382  heap->max_used_size = heap->pool_size - heap->available_size;
383 
384  /* remove next_ptr from free list */
385  RT_DEBUG_LOG(RT_DEBUG_MEMHEAP,
386  ("remove block: block[0x%08x], next_free 0x%08x, prev_free 0x%08x",
387  next_ptr,
388  next_ptr->next_free,
389  next_ptr->prev_free));
390 
391  next_ptr->next_free->prev_free = next_ptr->prev_free;
392  next_ptr->prev_free->next_free = next_ptr->next_free;
393  next_ptr->next->prev = next_ptr->prev;
394  next_ptr->prev->next = next_ptr->next;
395 
396  /* build a new one on the right place */
397  next_ptr = (struct rt_memheap_item *)((char *)ptr + newsize);
398 
399  RT_DEBUG_LOG(RT_DEBUG_MEMHEAP,
400  ("new free block: block[0x%08x] nextm[0x%08x] prevm[0x%08x]",
401  next_ptr,
402  next_ptr->next,
403  next_ptr->prev));
404 
405  /* mark the new block as a memory block and freed. */
406  next_ptr->magic = RT_MEMHEAP_MAGIC;
407 
408  /* put the pool pointer into the new block. */
409  next_ptr->pool_ptr = heap;
410 
411  next_ptr->prev = header_ptr;
412  next_ptr->next = header_ptr->next;
413  header_ptr->next->prev = next_ptr;
414  header_ptr->next = next_ptr;
415 
416  /* insert next_ptr to free list */
417  next_ptr->next_free = heap->free_list->next_free;
418  next_ptr->prev_free = heap->free_list;
419  heap->free_list->next_free->prev_free = next_ptr;
420  heap->free_list->next_free = next_ptr;
421  RT_DEBUG_LOG(RT_DEBUG_MEMHEAP, ("new ptr: next_free 0x%08x, prev_free 0x%08x",
422  next_ptr->next_free,
423  next_ptr->prev_free));
424 
425  /* release lock */
426  rt_sem_release(&(heap->lock));
427 
428  return ptr;
429  }
430  }
431 
432  /* release lock */
433  rt_sem_release(&(heap->lock));
434 
435  /* re-allocate a memory block */
436  new_ptr = (void *)rt_memheap_alloc(heap, newsize);
437  if (new_ptr != RT_NULL)
438  {
439  rt_memcpy(new_ptr, ptr, oldsize < newsize ? oldsize : newsize);
440  rt_memheap_free(ptr);
441  }
442 
443  return new_ptr;
444  }
445 
446  /* don't split when there is less than one node space left */
447  if (newsize + RT_MEMHEAP_SIZE + RT_MEMHEAP_MINIALLOC >= oldsize)
448  return ptr;
449 
450  /* lock memheap */
451  result = rt_sem_take(&(heap->lock), RT_WAITING_FOREVER);
452  if (result != RT_EOK)
453  {
454  rt_set_errno(result);
455 
456  return RT_NULL;
457  }
458 
459  /* split the block. */
460  new_ptr = (struct rt_memheap_item *)
461  (((rt_uint8_t *)header_ptr) + newsize + RT_MEMHEAP_SIZE);
462 
463  RT_DEBUG_LOG(RT_DEBUG_MEMHEAP,
464  ("split: block[0x%08x] nextm[0x%08x] prevm[0x%08x] to new[0x%08x]\n",
465  header_ptr,
466  header_ptr->next,
467  header_ptr->prev,
468  new_ptr));
469 
470  /* mark the new block as a memory block and freed. */
471  new_ptr->magic = RT_MEMHEAP_MAGIC;
472  /* put the pool pointer into the new block. */
473  new_ptr->pool_ptr = heap;
474 
475  /* break down the block list */
476  new_ptr->prev = header_ptr;
477  new_ptr->next = header_ptr->next;
478  header_ptr->next->prev = new_ptr;
479  header_ptr->next = new_ptr;
480 
481  /* determine if the block can be merged with the next neighbor. */
482  if (!RT_MEMHEAP_IS_USED(new_ptr->next))
483  {
484  struct rt_memheap_item *free_ptr;
485 
486  /* merge block with next neighbor. */
487  free_ptr = new_ptr->next;
488  heap->available_size = heap->available_size - MEMITEM_SIZE(free_ptr);
489 
490  RT_DEBUG_LOG(RT_DEBUG_MEMHEAP,
491  ("merge: right node 0x%08x, next_free 0x%08x, prev_free 0x%08x\n",
492  header_ptr, header_ptr->next_free, header_ptr->prev_free));
493 
494  free_ptr->next->prev = new_ptr;
495  new_ptr->next = free_ptr->next;
496 
497  /* remove free ptr from free list */
498  free_ptr->next_free->prev_free = free_ptr->prev_free;
499  free_ptr->prev_free->next_free = free_ptr->next_free;
500  }
501 
502  /* insert the split block to free list */
503  new_ptr->next_free = heap->free_list->next_free;
504  new_ptr->prev_free = heap->free_list;
505  heap->free_list->next_free->prev_free = new_ptr;
506  heap->free_list->next_free = new_ptr;
507  RT_DEBUG_LOG(RT_DEBUG_MEMHEAP, ("new free ptr: next_free 0x%08x, prev_free 0x%08x\n",
508  new_ptr->next_free,
509  new_ptr->prev_free));
510 
511  /* increment the available byte count. */
512  heap->available_size = heap->available_size + MEMITEM_SIZE(new_ptr);
513 
514  /* release lock */
515  rt_sem_release(&(heap->lock));
516 
517  /* return the old memory block */
518  return ptr;
519 }

522 void rt_memheap_free(void *ptr)
523 {
524  rt_err_t result;
525  struct rt_memheap *heap;
526  struct rt_memheap_item *header_ptr, *new_ptr;
527  rt_uint32_t insert_header;
528 
529  /* NULL check */
530  if (ptr == RT_NULL) return;
531 
532  /* set initial status as OK */
533  insert_header = 1;
534  new_ptr = RT_NULL;
535  header_ptr = (struct rt_memheap_item *)
536  ((rt_uint8_t *)ptr - RT_MEMHEAP_SIZE);
537 
538  RT_DEBUG_LOG(RT_DEBUG_MEMHEAP, ("free memory: memory[0x%08x], block[0x%08x]\n",
539  ptr, header_ptr));
540 
541  /* check magic */
542  RT_ASSERT((header_ptr->magic & RT_MEMHEAP_MASK) == RT_MEMHEAP_MAGIC);
543  RT_ASSERT(header_ptr->magic & RT_MEMHEAP_USED);
544  /* check whether this block of memory has been over-written. */
545  RT_ASSERT((header_ptr->next->magic & RT_MEMHEAP_MASK) == RT_MEMHEAP_MAGIC);
546 
547  /* get pool ptr */
548  heap = header_ptr->pool_ptr;
549 
550  /* lock memheap */
551  result = rt_sem_take(&(heap->lock), RT_WAITING_FOREVER);
552  if (result != RT_EOK)
553  {
554  rt_set_errno(result);
555 
556  return ;
557  }
558 
559  /* Mark the memory as available. */
560  header_ptr->magic &= ~RT_MEMHEAP_USED;
561  /* Adjust the available number of bytes. */
562  heap->available_size = heap->available_size + MEMITEM_SIZE(header_ptr);
563 
564  /* Determine if the block can be merged with the previous neighbor. */
565  if (!RT_MEMHEAP_IS_USED(header_ptr->prev))
566  {
567  RT_DEBUG_LOG(RT_DEBUG_MEMHEAP, ("merge: left node 0x%08x\n",
568  header_ptr->prev));
569 
570  /* adjust the available number of bytes. */
571  heap->available_size = heap->available_size + RT_MEMHEAP_SIZE;
572 
573  /* yes, merge block with previous neighbor. */
574  (header_ptr->prev)->next = header_ptr->next;
575  (header_ptr->next)->prev = header_ptr->prev;
576 
577  /* move header pointer to previous. */
578  header_ptr = header_ptr->prev;
579  /* don't insert header to free list */
580  insert_header = 0;
581  }
582 
583  /* determine if the block can be merged with the next neighbor. */
584  if (!RT_MEMHEAP_IS_USED(header_ptr->next))
585  {
586  /* adjust the available number of bytes. */
587  heap->available_size = heap->available_size + RT_MEMHEAP_SIZE;
588 
589  /* merge block with next neighbor. */
590  new_ptr = header_ptr->next;
591 
592  RT_DEBUG_LOG(RT_DEBUG_MEMHEAP,
593  ("merge: right node 0x%08x, next_free 0x%08x, prev_free 0x%08x\n",
594  new_ptr, new_ptr->next_free, new_ptr->prev_free));
595 
596  new_ptr->next->prev = header_ptr;
597  header_ptr->next = new_ptr->next;
598 
599  /* remove new ptr from free list */
600  new_ptr->next_free->prev_free = new_ptr->prev_free;
601  new_ptr->prev_free->next_free = new_ptr->next_free;
602  }
603 
604  if (insert_header)
605  {
606  /* no left merge, insert to free list */
607  header_ptr->next_free = heap->free_list->next_free;
608  header_ptr->prev_free = heap->free_list;
609  heap->free_list->next_free->prev_free = header_ptr;
610  heap->free_list->next_free = header_ptr;
611 
612  RT_DEBUG_LOG(RT_DEBUG_MEMHEAP,
613  ("insert to free list: next_free 0x%08x, prev_free 0x%08x\n",
614  header_ptr->next_free, header_ptr->prev_free));
615  }
616 
617  /* release lock */
618  rt_sem_release(&(heap->lock));
619 }
