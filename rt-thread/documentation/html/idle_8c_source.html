73 /* Return whether there is defunctional thread to be deleted. */
74 rt_inline int _has_defunct_thread(void)
75 {
76  /* The rt_list_isempty has prototype of "int rt_list_isempty(const rt_list_t *l)".
77  * So the compiler has a good reason that the rt_thread_defunct list does
78  * not change within rt_thread_idle_excute thus optimize the "while" loop
79  * into a "if".
80  *
81  * So add the volatile qualifier here. */
82  const volatile rt_list_t *l = (const volatile rt_list_t *)&rt_thread_defunct;
83 
84  return l->next != l;
85 }

87 /**
88  * @ingroup Thread
89  * @brief 系统空闲执行函数
90  *
91  * 当系统空闲时将在后台执行该函数。
92  */
93 void rt_thread_idle_excute(void)
94 {
95  /* Loop until there is no dead thread. So one call to rt_thread_idle_excute
96  * will do all the cleanups. */
97  while (_has_defunct_thread())
98  {
99  rt_base_t lock;
100  rt_thread_t thread;
101 #ifdef RT_USING_MODULE
102  rt_module_t module = RT_NULL;
103 #endif
104  RT_DEBUG_NOT_IN_INTERRUPT;
105 
106  /* disable interrupt */
107  lock = rt_hw_interrupt_disable();
108 
109  /* re-check whether list is empty */
110  if (_has_defunct_thread())
111  {
112  /* get defunct thread */
113  thread = rt_list_entry(rt_thread_defunct.next,
114  struct rt_thread,
115  tlist);
116 #ifdef RT_USING_MODULE
117  /* get thread's parent module */
118  module = (rt_module_t)thread->module_id;
119 
120  /* if the thread is module's main thread */
121  if (module != RT_NULL && module->module_thread == thread)
122  {
123  /* detach module's main thread */
124  module->module_thread = RT_NULL;
125  }
126 #endif
127  /* remove defunct thread */
128  rt_list_remove(&(thread->tlist));
129 
130  /* lock scheduler to prevent scheduling in cleanup function. */
131  rt_enter_critical();
132 
133  /* invoke thread cleanup */
134  if (thread->cleanup != RT_NULL)
135  thread->cleanup(thread);
136 
137 #ifdef RT_USING_SIGNALS
138  rt_thread_free_sig(thread);
139 #endif
140 
141  /* if it's a system object, not delete it */
142  if (rt_object_is_systemobject((rt_object_t)thread) == RT_TRUE)
143  {
144  /* unlock scheduler */
145  rt_exit_critical();
146 
147  /* enable interrupt */
148  rt_hw_interrupt_enable(lock);
149 
150  return;
151  }
152 
153  /* unlock scheduler */
154  rt_exit_critical();
155  }
156  else
157  {
158  /* enable interrupt */
159  rt_hw_interrupt_enable(lock);
160 
161  /* may the defunct thread list is removed by others, just return */
162  return;
163  }
164 
165  /* enable interrupt */
166  rt_hw_interrupt_enable(lock);
167 
168 #ifdef RT_USING_HEAP
169  /* release thread's stack */
170  RT_KERNEL_FREE(thread->stack_addr);
171  /* delete thread object */
172  rt_object_delete((rt_object_t)thread);
173 #endif
174  }
175 }

177 static void rt_thread_idle_entry(void *parameter)
178 {
179  while (1)
180  {
181 #ifdef RT_USING_IDLE_HOOK
182  if (rt_thread_idle_hook != RT_NULL)
183  {
184  rt_thread_idle_hook();
185  }
186 #endif
187 
188  rt_thread_idle_excute();
189  }
